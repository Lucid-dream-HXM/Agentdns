BASE_URL = "http://127.0.0.1:8000/api/v1"

def auth_headers(api_key: str) -> Dict[str, str]:
    return {
        "Authorization": f"Bearer {api_key}",
        "Content-Type": "application/json"
    }


def search_services(api_key: str, task: Dict[str, Any]) -> List[Dict[str, Any]]:
    payload = {
        "query": task["query"],
        "category": task["category"],
        "limit": 10,
        "return_tool_format": False,
        "include_trust": True,
        "sort_by": "balanced"
    }

    resp = requests.post(
        f"{BASE_URL}/client/discovery/search",
        headers=auth_headers(api_key),

        timeout=30
    )
    resp.raise_for_status()
    data = resp.json()
    return data.get("services", [])


def choose_service_for_warmup(profile: str, candidates: List[Dict[str, Any]], top_k: int) -> Dict[str, Any]:
    """
    预热阶段的选择策略

    说明：
    - 不追求最优实验选择
    - 目标是让不同服务都获得一定曝光，同时高质量服务逐步积累更好评价
    """
    visible = candidates[:top_k]
    if not visible:
        raise RuntimeError("Top-K 候选为空")

    if profile == "quality_first":
        # 预热时优先高 trust / 高价 / specialist
        def score(svc: Dict[str, Any]) -> float:
            trust = float(svc.get("trust_score") or 0.0)
            price = float(svc.get("price_per_unit") or 0.0)
            role_type = (svc.get("capabilities") or {}).get("role_type", "")
            specialist_bonus = 6.0 if role_type == "specialist" else 0.0
            return trust + specialist_bonus - price * 12.0

        ranked = sorted(visible, key=score, reverse=True)
        return ranked[0]

    # 成本型 agent 偶尔故意探索低价服务
    def score_cost(svc: Dict[str, Any]) -> float:
        trust = float(svc.get("trust_score") or 0.0)
        price = float(svc.get("price_per_unit") or 0.0)
        return (10.0 - price * 100.0) + trust * 0.15

    ranked = sorted(visible, key=score_cost, reverse=True)
    return ranked[0]


def call_service(api_key: str, service: Dict[str, Any], task: Dict[str, Any]) -> requests.Response:
    capabilities = service.get("capabilities") or {}
    service_key = capabilities.get("service_key")
    if not service_key:
        raise RuntimeError(f"服务 {service.get('name')} 缺少 capabilities.service_key")

    payload = {
        "agentdns_url": service["agentdns_uri"],
        "input_data": {
            "service_key": service_key,
            "task_input": {
                "text": task["text"]
            },
            "stage": "stable_growth"
        },
        "method": "POST"
    }

    return requests.post(
        f"{BASE_URL}/client/services/call",
        headers=auth_headers(api_key),
        json=payload,
        timeout=60
    )


def extract_usage_id(resp: requests.Response):
    value = resp.headers.get("X-AgentDNS-Usage-ID")
    if not value:
        return None
    try:
        return int(value)
    except ValueError:
        return None


def build_review_payload(usage_id: int, service: Dict[str, Any], call_data: Dict[str, Any], profile: str) -> Dict[str, Any]:
    outcome = call_data.get("status", "fail")
    price = float(service.get("price_per_unit") or 0.0)

    if outcome == "success":
        task_fit = 5
        output_quality = 4 if profile == "quality_first" else 3
        protocol_adherence = 5
        would_reuse = True
    elif outcome == "partial":
        task_fit = 3
        output_quality = 3
        protocol_adherence = 4
        would_reuse = False
    else:
        task_fit = 1
        output_quality = 1
        protocol_adherence = 2
        would_reuse = False

    if price <= 0.03:
        cost_satisfaction = 5
    elif price <= 0.06:
        cost_satisfaction = 4
    elif price <= 0.08:
        cost_satisfaction = 3
    else:
        cost_satisfaction = 2

    return {
        "usage_id": usage_id,
        "outcome": outcome,
        "task_fit": task_fit,
        "output_quality": output_quality,
        "protocol_adherence": protocol_adherence,
        "would_reuse": would_reuse,
        "cost_satisfaction": cost_satisfaction,
        "feedback_text": f"warmup review for {service.get('name')}",
        "evidence": {
            "profile": profile,
            "warmup": True
        }
    }


def submit_review(api_key: str, payload: Dict[str, Any]) -> None:
    resp = requests.post(
        f"{BASE_URL}/reviews/",
        headers=auth_headers(api_key),
        json=payload,
        timeout=30
    )
    resp.raise_for_status()


def main() -> None:
    print("=" * 72)
    print("开始执行 trust warm-up")
    print("=" * 72)

    for idx, task in enumerate(WARMUP_TASKS):
        agent_cfg = AGENTS[idx % len(AGENTS)]
        profile = agent_cfg["profile"]
        api_key = agent_cfg["api_key"]

        try:
            candidates = search_services(api_key, task)
            if not candidates:
                print(f"[WARMUP] task={task['task_id']} 无候选")
                continue

            selected = choose_service_for_warmup(profile, candidates, task["top_k"])
            resp = call_service(api_key, selected, task)
            call_data = resp.json()

            usage_id = extract_usage_id(resp)
            if usage_id is None:
                print(f"[WARMUP] task={task['task_id']} missing_usage_id")
                continue

            review_payload = build_review_payload(
                usage_id=usage_id,
                service=selected,
                call_data=call_data,
                profile=profile
            )
            submit_review(api_key, review_payload)

            print(
                f"[WARMUP] task={task['task_id']} "
                f"profile={profile} "
                f"service={selected['name']} "
                f"status={call_data.get('status')}"
            )

        except Exception as e:
            print(f"[WARMUP-ERROR] task={task['task_id']} error={e}")

        time.sleep(0.3)

    print("=" * 72)
    print("trust warm-up 完成")
    print("=" * 72)


if __name__ == "__main__":
    main()

from __future__ import annotations

import csv
from pathlib import Path
from typing import Dict, List

import matplotlib.pyplot as plt


INPUT_PATH = Path("experiments/outputs/runs/v1_vs_v3_summary.csv")
FIG_DIR = Path("experiments/outputs/figures")
FIG_DIR.mkdir(parents=True, exist_ok=True)


def load_summary_rows() -> List[Dict[str, str]]:
    if not INPUT_PATH.exists():
        raise FileNotFoundError(f"汇总文件不存在: {INPUT_PATH}")

    with INPUT_PATH.open("r", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        return list(reader)


def build_single_bar(values: Dict[str, float], ylabel: str, title: str, output_name: str) -> None:
    versions = list(values.keys())
    data = [values[v] for v in versions]

    plt.figure(figsize=(6, 4))
    plt.bar(versions, data)
    plt.ylabel(ylabel)
    plt.title(title)
    plt.tight_layout()
    plt.savefig(FIG_DIR / output_name, dpi=180)
    plt.close()

    rows = load_summary_rows()
    if not rows:
        raise ValueError("汇总文件为空，无法出图")

    success_rate = {row["version"]: float(row["task_success_rate"]) for row in rows}
    avg_utility = {row["version"]: float(row["avg_oracle_utility"]) for row in rows}
    avg_cost = {row["version"]: float(row["avg_total_price"]) for row in rows}

        title="V1 vs V3 Average Total Cost",
