"""
V1 vs V3 最小配对对比实验执行器

当前阶段目标：
1. 使用同一批中等任务实例
2. 在相同随机种子下分别运行 V1 / V3
3. 输出统一 jsonl 对比日志

定义：
- V1: 无 trust
  * discovery: include_trust = false
  * sort_by = relevance
  * 本地候选选择不使用 trust 字段

- V3: 静态 trust
  * discovery: include_trust = true
  * sort_by = balanced
  * 本地候选选择使用 trust 字段
"""

from __future__ import annotations

import json
import random
import time
from pathlib import Path
from typing import Any, Dict, List, Optional

import requests

from experiments.evaluators.task_success import evaluate_medium_task_success
from experiments.evaluators.oracle_utility import evaluate_medium_oracle_utility


BASE_URL = "http://127.0.0.1:8000/api/v1"
TEMPLATE_PATH = Path("experiments/configs/tasks/medium_task_templates.json")
OUTPUT_PATH = Path("experiments/outputs/runs/v1_vs_v3_runs.jsonl")

AGENTS = [
    {

        "api_key": "agent_exp_quality_01",

    {

def auth_headers(api_key: str) -> Dict[str, str]:
    return {
        "Authorization": f"Bearer {api_key}",
        "Content-Type": "application/json"
    }


def append_run_log(record: Dict[str, Any]) -> None:
    OUTPUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    with OUTPUT_PATH.open("a", encoding="utf-8") as f:
        f.write(json.dumps(record, ensure_ascii=False) + "\n")


def load_templates() -> List[Dict[str, Any]]:
    with TEMPLATE_PATH.open("r", encoding="utf-8") as f:
        return json.load(f)


def instantiate_medium_tasks(total_count: int = 8, seed: int = 20260414) -> List[Dict[str, Any]]:
    rng = random.Random(seed)
    templates = load_templates()

    instances: List[Dict[str, Any]] = []
    for idx in range(total_count):
        tpl = templates[idx % len(templates)]
        text = rng.choice(tpl["input_candidates"])
        instances.append({
            "task_instance_id": f"pair_{idx + 1:04d}",
            "template_id": tpl["template_id"],
            "domain": tpl["domain"],
            "complexity": tpl["complexity"],
            "workflow_type": tpl["workflow_type"],
            "query_step_1": tpl["query_step_1"],
            "category_step_1": tpl["category_step_1"],
            "query_step_2": tpl["query_step_2"],
            "category_step_2": tpl["category_step_2"],
            "top_k_step_1": tpl["top_k_step_1"],
            "top_k_step_2": tpl["top_k_step_2"],
            "hard_constraints": tpl.get("hard_constraints", {}),
            "soft_preferences": tpl.get("soft_preferences", {}),
            "text": text,
            "seed": seed
        })

    rng.shuffle(instances)
    return instances


def search_services(
    api_key: str,
    query: str,
    category: str,
    max_price: Optional[float],
    version: str
) -> List[Dict[str, Any]]:
    payload = {
        "query": query,
        "category": category,
        "limit": 10,
        "return_tool_format": False
    }

    if max_price is not None:
        payload["max_price"] = max_price

    if version == "V1":
        payload["include_trust"] = False
        payload["sort_by"] = "relevance"
    elif version == "V3":
        payload["include_trust"] = True
        payload["sort_by"] = "balanced"
    else:
        raise ValueError(f"未知版本: {version}")

    resp = requests.post(
        f"{BASE_URL}/client/discovery/search",
        headers=auth_headers(api_key),
        json=payload,
        timeout=30
    )
    resp.raise_for_status()
    data = resp.json()
    return data.get("services", [])


def choose_service(
    profile: str,
    candidates: List[Dict[str, Any]],
    top_k: int,
    version: str
) -> Dict[str, Any]:
    visible = candidates[:top_k]
    if not visible:
        raise RuntimeError("Top-K 候选为空")

    def score_v1_quality(svc: Dict[str, Any]) -> float:
        price = float(svc.get("price_per_unit") or 0.0)
        role_type = (svc.get("capabilities") or {}).get("role_type", "")
        specialist_bonus = 8.0 if role_type == "specialist" else 0.0
        return specialist_bonus - price * 20.0

    def score_v1_cost(svc: Dict[str, Any]) -> float:
        price = float(svc.get("price_per_unit") or 0.0)
        return 10.0 - price * 100.0

    def score_v3_quality(svc: Dict[str, Any]) -> float:
        trust = float(svc.get("trust_score") or 0.0)
        price = float(svc.get("price_per_unit") or 0.0)
        role_type = (svc.get("capabilities") or {}).get("role_type", "")
        specialist_bonus = 8.0 if role_type == "specialist" else 0.0
        return trust + specialist_bonus - price * 20.0

    def score_v3_cost(svc: Dict[str, Any]) -> float:
        trust = float(svc.get("trust_score") or 0.0)
        price = float(svc.get("price_per_unit") or 0.0)
        return (10.0 - price * 100.0) + trust * 0.2

    if version == "V1":
        if profile == "quality_first":
            ranked = sorted(visible, key=score_v1_quality, reverse=True)
        elif profile == "cost_first":
            ranked = sorted(visible, key=score_v1_cost, reverse=True)
        else:
            ranked = visible
    elif version == "V3":
        if profile == "quality_first":
            ranked = sorted(visible, key=score_v3_quality, reverse=True)
        elif profile == "cost_first":
            ranked = sorted(visible, key=score_v3_cost, reverse=True)
        else:
            ranked = visible
    else:
        raise ValueError(f"未知版本: {version}")

    return ranked[0]


def call_service(api_key: str, service: Dict[str, Any], text: str) -> requests.Response:
    capabilities = service.get("capabilities") or {}
    service_key = capabilities.get("service_key")
    if not service_key:
        raise RuntimeError(f"服务 {service.get('name')} 缺少 capabilities.service_key")

    payload = {
        "agentdns_url": service["agentdns_uri"],
        "input_data": {
            "service_key": service_key,
            "task_input": {
                "text": text
            },
            "stage": "cold_start"
        },
        "method": "POST"
    }

    return requests.post(
        f"{BASE_URL}/client/services/call",
        headers=auth_headers(api_key),
        json=payload,
        timeout=60
    )


def extract_usage_id(resp: requests.Response) -> Optional[int]:
    value = resp.headers.get("X-AgentDNS-Usage-ID")
    if not value:
        return None
    try:
        return int(value)
    except ValueError:
        return None


def build_review_payload(usage_id: int, service: Dict[str, Any], call_data: Dict[str, Any], profile: str) -> Dict[str, Any]:
    outcome = call_data.get("status", "fail")
    price = float(service.get("price_per_unit") or 0.0)

    if outcome == "success":
        task_fit = 5
        output_quality = 4 if profile == "quality_first" else 3
        protocol_adherence = 5
        would_reuse = True
    elif outcome == "partial":
        task_fit = 3
        output_quality = 3
        protocol_adherence = 4
        would_reuse = False
    else:
        task_fit = 1
        output_quality = 1
        protocol_adherence = 2
        would_reuse = False

    if price <= 0.03:
        cost_satisfaction = 5
    elif price <= 0.06:
        cost_satisfaction = 4
    elif price <= 0.08:
        cost_satisfaction = 3
    else:
        cost_satisfaction = 2

    return {
        "usage_id": usage_id,
        "outcome": outcome,
        "task_fit": task_fit,
        "output_quality": output_quality,
        "protocol_adherence": protocol_adherence,
        "would_reuse": would_reuse,
        "cost_satisfaction": cost_satisfaction,
        "feedback_text": f"{profile} auto review",
        "evidence": {
            "visible_status": outcome,
            "profile": profile
        }
    }


def submit_review(api_key: str, payload: Dict[str, Any]) -> Dict[str, Any]:
    resp = requests.post(
        f"{BASE_URL}/reviews/",
        headers=auth_headers(api_key),
        json=payload,
        timeout=30
    )
    resp.raise_for_status()
    return resp.json()


def execute_step(
    api_key: str,
    profile: str,
    query: str,
    category: str,
    top_k: int,
    input_text: str,
    max_price: Optional[float],
    version: str
) -> Dict[str, Any]:
    step_log: Dict[str, Any] = {
        "query": query,
        "category": category,
        "top_k": top_k,
        "search_candidates_count": 0,
        "visible_candidates": [],
        "selected_service_name": None,
        "selected_service_id": None,
        "selected_service_price": None,
        "selected_service_trust_before": None,
        "call_http_status": None,
        "call_status": None,
        "request_id": None,
        "usage_id": None,
        "review_id": None,
        "trust_score_after": None,
        "objective_score_after": None,
        "subjective_score_after": None,
        "rating_count_after": None,
        "error": None
    }

    try:
        candidates = search_services(api_key, query, category, max_price, version)
        step_log["search_candidates_count"] = len(candidates)
        step_log["visible_candidates"] = [
            {
                "name": svc.get("name"),
                "id": svc.get("id"),
                "price_per_unit": svc.get("price_per_unit"),
                "trust_score": svc.get("trust_score")
            }
            for svc in candidates[:top_k]
        ]

        if not candidates:
            step_log["error"] = "no_candidates"
            return step_log

        selected = choose_service(profile, candidates, top_k, version)
        step_log["selected_service_name"] = selected.get("name")
        step_log["selected_service_id"] = selected.get("id")
        step_log["selected_service_price"] = selected.get("price_per_unit")
        step_log["selected_service_trust_before"] = selected.get("trust_score")

        resp = call_service(api_key, selected, input_text)
        step_log["call_http_status"] = resp.status_code

        try:
            call_data = resp.json()
        except Exception:
            step_log["error"] = f"call_json_parse_error: {resp.text[:200]}"
            return step_log

        step_log["call_status"] = call_data.get("status")
        step_log["request_id"] = call_data.get("request_id")

        usage_id = extract_usage_id(resp)
        step_log["usage_id"] = usage_id
        if usage_id is None:
            step_log["error"] = "missing_usage_id"
            return step_log

        review_payload = build_review_payload(usage_id, selected, call_data, profile)
        review_data = submit_review(api_key, review_payload)
        step_log["review_id"] = review_data.get("id")

        # 只有 V3 会真正使用 trust 字段参与下一轮选择，但两边都记录下来
        summary_resp = requests.get(
            f"{BASE_URL}/reviews/services/{selected['id']}/summary",
            headers=auth_headers(api_key),
            timeout=30
        )
        summary_resp.raise_for_status()
        trust_summary = summary_resp.json()

        step_log["trust_score_after"] = trust_summary.get("trust_score")
        step_log["objective_score_after"] = trust_summary.get("objective_score")
        step_log["subjective_score_after"] = trust_summary.get("subjective_score")
        step_log["rating_count_after"] = trust_summary.get("rating_count")

        if "result" in call_data and isinstance(call_data["result"], dict):
            result_obj = call_data["result"]
            if "text" in result_obj:
                step_log["next_input_text"] = result_obj["text"]
            elif "structured_data" in result_obj:
                step_log["next_input_text"] = json.dumps(result_obj["structured_data"], ensure_ascii=False)
            else:
                step_log["next_input_text"] = json.dumps(result_obj, ensure_ascii=False)
        else:
            step_log["next_input_text"] = input_text

    except Exception as e:
        step_log["error"] = str(e)

    return step_log


def run_single_task(task: Dict[str, Any], agent_cfg: Dict[str, Any], version: str) -> None:
    api_key = agent_cfg["api_key"]
    profile = agent_cfg["profile"]
    max_total_price = task.get("hard_constraints", {}).get("max_total_price")

    run_record: Dict[str, Any] = {

        "task_instance_id": task["task_instance_id"],
        "template_id": task["template_id"],
        "domain": task["domain"],
        "complexity": task["complexity"],
        "workflow_type": task["workflow_type"],
        "agent_name": agent_cfg["agent_name"],
        "profile": profile,
        "hard_constraints": task.get("hard_constraints", {}),
        "soft_preferences": task.get("soft_preferences", {}),
        "step1": {},
        "step2": {},
        "total_price_estimate": 0.0,
        "task_success": None,
        "failure_reason": None,
        "oracle_utility": None,
        "oracle_quality_score": None,
        "oracle_cost_score": None,
        "oracle_latency_score": None,
        "oracle_constraint_score": None,
        "oracle_missing": None,
        "error": None
    }

    try:
        step1 = execute_step(

            input_text=task["text"],

        )
        run_record["step1"] = step1

        total_price = float(step1.get("selected_service_price") or 0.0) + float(step2.get("selected_service_price") or 0.0)
        run_record["total_price_estimate"] = round(total_price, 6)

        run_record.update(evaluate_medium_task_success(run_record))
        run_record.update(evaluate_medium_oracle_utility(run_record))

    except Exception as e:
        run_record["error"] = str(e)

    append_run_log(run_record)


def main() -> None:
    if OUTPUT_PATH.exists():
        OUTPUT_PATH.unlink()

    tasks = instantiate_medium_tasks(total_count=8, seed=20260414)

    # 先跑 V1，再重置，再跑 V3 —— 由你手动在两次运行之间执行 reset 脚本

            print("=" * 72)

if __name__ == "__main__":

V1 vs V3 最小对比结果汇总器

from __future__ import annotations

import csv

def load_records(path: Path) -> List[Dict[str, Any]]:
    if not path.exists():
        raise FileNotFoundError(f"运行日志不存在: {path}")

    records: List[Dict[str, Any]] = []
    with path.open("r", encoding="utf-8") as f:

def export_csv(rows: List[Dict[str, Any]], path: Path) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    fieldnames = [
        "version",
        "task_count",
        "task_success_count",
        "task_success_rate",
        "avg_oracle_utility",
        "avg_total_price",
        "avg_step1_price",
        "avg_step2_price"
    ]

    with path.open("w", encoding="utf-8", newline="") as f:
        writer = csv.DictWriter(f, fieldnames=fieldnames)
        writer.writeheader()
        for row in rows:
            writer.writerow(row)


def summarize_by_version(records: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    versions = sorted(set(r.get("version") for r in records if r.get("version")))
    rows: List[Dict[str, Any]] = []

    for version in versions:
        subset = [r for r in records if r.get("version") == version]
        if not subset:
            continue

        task_count = len(subset)
        success_count = sum(1 for r in subset if r.get("task_success") is True)
        success_rate = success_count / task_count if task_count else 0.0

        utility_values = [float(r["oracle_utility"]) for r in subset if r.get("oracle_utility") is not None]
        avg_utility = sum(utility_values) / len(utility_values) if utility_values else 0.0

        total_prices = [float(r.get("total_price_estimate") or 0.0) for r in subset]
        avg_total_price = sum(total_prices) / len(total_prices) if total_prices else 0.0

        step1_prices = [float((r.get("step1") or {}).get("selected_service_price") or 0.0) for r in subset]
        step2_prices = [float((r.get("step2") or {}).get("selected_service_price") or 0.0) for r in subset]

        rows.append({
            "version": version,
            "task_count": task_count,
            "task_success_count": success_count,
            "task_success_rate": round(success_rate, 4),
            "avg_oracle_utility": round(avg_utility, 4),
            "avg_total_price": round(sum(total_prices) / len(total_prices), 4) if total_prices else 0.0,
            "avg_step1_price": round(sum(step1_prices) / len(step1_prices), 4) if step1_prices else 0.0,
            "avg_step2_price": round(sum(step2_prices) / len(step2_prices), 4) if step2_prices else 0.0
        })

    return rows


def main() -> None:
    records = load_records(INPUT_PATH)
    rows = summarize_by_version(records)
    export_csv(rows, CSV_OUTPUT_PATH)

    print("=" * 72)
    print("V1 vs V3 最小对比结果")
    for row in rows:
        print(
            f"[{row['version']}] "
            f"task_success_rate={row['task_success_rate']} "
            f"avg_oracle_utility={row['avg_oracle_utility']} "
            f"avg_total_price={row['avg_total_price']}"
        )
    print(f"CSV 输出: {CSV_OUTPUT_PATH}")
    print("=" * 72)


if __name__ == "__main__":
    main()
