```python
"""
生成论文可直接使用的结果表格与文字描述

输入：
- experiments/outputs/runs/v1_vs_v3_summary.csv

输出：
- experiments/outputs/reports/v1_v3_result_table.md
- experiments/outputs/reports/v1_v3_result_table.tex
- experiments/outputs/reports/v1_v3_result_narrative.md
"""

from __future__ import annotations

import csv
from pathlib import Path
from typing import Dict, List


INPUT_PATH = Path("experiments/outputs/runs/v1_vs_v3_summary.csv")
REPORT_DIR = Path("experiments/outputs/reports")

MD_TABLE_PATH = REPORT_DIR / "v1_v3_result_table.md"
TEX_TABLE_PATH = REPORT_DIR / "v1_v3_result_table.tex"
NARRATIVE_PATH = REPORT_DIR / "v1_v3_result_narrative.md"


def load_rows() -> List[Dict[str, str]]:
    """
    加载 V1/V3 汇总结果
    """
    if not INPUT_PATH.exists():
        raise FileNotFoundError(f"未找到输入文件: {INPUT_PATH}")

    with INPUT_PATH.open("r", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        rows = list(reader)

    if not rows:
        raise ValueError("输入 CSV 为空")

    return rows


def get_version_map(rows: List[Dict[str, str]]) -> Dict[str, Dict[str, str]]:
    """
    将结果按版本名映射
    """
    mapping: Dict[str, Dict[str, str]] = {}
    for row in rows:
        version = row.get("version")
        if version:
            mapping[version] = row
    return mapping


def f4(value: str | float) -> str:
    """
    格式化为 4 位小数
    """
    return f"{float(value):.4f}"


def f1_percent(ratio: str | float) -> str:
    """
    将比例转成百分数，保留 1 位小数
    """
    return f"{float(ratio) * 100:.1f}\\%"


def build_md_table(v1: Dict[str, str], v3: Dict[str, str]) -> str:
    """
    生成 Markdown 表格
    """
    v1_success = float(v1["task_success_rate"])
    v3_success = float(v3["task_success_rate"])
    v1_utility = float(v1["avg_oracle_utility"])
    v3_utility = float(v3["avg_oracle_utility"])
    v1_cost = float(v1["avg_total_price"])
    v3_cost = float(v3["avg_total_price"])

    success_delta = (v3_success - v1_success) / v1_success if v1_success > 0 else 0.0
    utility_delta = (v3_utility - v1_utility) / v1_utility if v1_utility > 0 else 0.0
    cost_delta = (v3_cost - v1_cost) / v1_cost if v1_cost > 0 else 0.0

    return f"""# V1 与 V3 最小对比实验结果表

| 指标 | V1 | V3 | 相对变化 |
|------|----|----|----------|
| 任务成功率 | {v1_success * 100:.1f}% | {v3_success * 100:.1f}% | {success_delta * 100:+.1f}% |
| 平均 Oracle Utility | {v1_utility:.4f} | {v3_utility:.4f} | {utility_delta * 100:+.1f}% |
| 平均总成本 | {v1_cost:.4f} | {v3_cost:.4f} | {cost_delta * 100:+.1f}% |

说明：
- V1：无 trust 的发现基线
- V3：静态 trust 版本
- 相对变化以 V1 为基线计算
"""
    

def build_tex_table(v1: Dict[str, str], v3: Dict[str, str]) -> str:
    """
    生成 LaTeX 表格
    """
    v1_success = float(v1["task_success_rate"])
    v3_success = float(v3["task_success_rate"])
    v1_utility = float(v1["avg_oracle_utility"])
    v3_utility = float(v3["avg_oracle_utility"])
    v1_cost = float(v1["avg_total_price"])
    v3_cost = float(v3["avg_total_price"])

    success_delta = (v3_success - v1_success) / v1_success if v1_success > 0 else 0.0
    utility_delta = (v3_utility - v1_utility) / v1_utility if v1_utility > 0 else 0.0
    cost_delta = (v3_cost - v1_cost) / v1_cost if v1_cost > 0 else 0.0

    return rf"""\begin{{table}}[htbp]
\centering
\caption{{V1 与 V3 在最小对比实验中的结果}}
\label{{tab:v1_v3_minimal_results}}
\begin{{tabular}}{{lccc}}
\hline
指标 & V1 & V3 & 相对变化 \\
\hline
任务成功率 & {v1_success * 100:.1f}\% & {v3_success * 100:.1f}\% & {success_delta * 100:+.1f}\% \\
平均 Oracle Utility & {v1_utility:.4f} & {v3_utility:.4f} & {utility_delta * 100:+.1f}\% \\
平均总成本 & {v1_cost:.4f} & {v3_cost:.4f} & {cost_delta * 100:+.1f}\% \\
\hline
\end{{tabular}}
\end{{table}}
"""


def build_narrative(v1: Dict[str, str], v3: Dict[str, str]) -> str:
    """
    生成论文风格的中文结果描述
    """
    v1_success = float(v1["task_success_rate"])
    v3_success = float(v3["task_success_rate"])
    v1_utility = float(v1["avg_oracle_utility"])
    v3_utility = float(v3["avg_oracle_utility"])
    v1_cost = float(v1["avg_total_price"])
    v3_cost = float(v3["avg_total_price"])

    success_delta = (v3_success - v1_success) / v1_success if v1_success > 0 else 0.0
    utility_delta = (v3_utility - v1_utility) / v1_utility if v1_utility > 0 else 0.0
    cost_delta = (v3_cost - v1_cost) / v1_cost if v1_cost > 0 else 0.0

    direction_success = "提升" if success_delta >= 0 else "下降"
    direction_utility = "提升" if utility_delta >= 0 else "下降"
    direction_cost = "下降" if cost_delta < 0 else "上升"

    return f"""# 论文结果描述草稿（V1 vs V3）

在扩展后的最小实验服务市场中，本文对无 trust 的发现基线（V1）与静态 trust 版本（V3）进行了对比实验。实验结果表明，V3 在任务成功率、选择效用与成本控制方面已开始表现出区别于基线方法的行为特征。

具体来看，V1 的任务成功率为 {v1_success * 100:.1f}% ，而 V3 的任务成功率为 {v3_success * 100:.1f}% ，相较于 V1 {direction_success}了 {abs(success_delta) * 100:.1f}% 。在选择层指标上，V1 的平均 Oracle Utility 为 {v1_utility:.4f} ，V3 为 {v3_utility:.4f} ，相较于基线 {direction_utility}了 {abs(utility_delta) * 100:.1f}% 。在成本方面，V1 的平均总成本为 {v1_cost:.4f} ，而 V3 的平均总成本为 {v3_cost:.4f} ，相较于基线 {direction_cost}了 {abs(cost_delta) * 100:.1f}% 。

从服务选择行为上看，V1 更倾向于直接采用发现结果中的高价高质量服务，而 V3 在已有信任信息的条件下，能够在“质量足够可用”的前提下更多选择低成本候选。因此，在当前小型竞争市场中，静态 trust 机制已经开始体现出一种更有平台意义的选择偏好：即不仅关注任务是否完成，也关注在多候选竞争下的成本—效用平衡。

这一结果说明，当候选服务之间存在较明显的质量分层与价格分层时，trust 机制可以对服务选择产生实质影响；而这种影响并不一定表现为单纯追求最高质量服务，而更可能体现为在保证任务成功的同时降低总体使用成本。
"""


def main() -> None:
    REPORT_DIR.mkdir(parents=True, exist_ok=True)

    rows = load_rows()
    version_map = get_version_map(rows)

    if "V1" not in version_map or "V3" not in version_map:
        raise ValueError("CSV 中必须同时包含 V1 和 V3 结果")

    v1 = version_map["V1"]
    v3 = version_map["V3"]

    md_table = build_md_table(v1, v3)
    tex_table = build_tex_table(v1, v3)
    narrative = build_narrative(v1, v3)

    MD_TABLE_PATH.write_text(md_table, encoding="utf-8")
    TEX_TABLE_PATH.write_text(tex_table, encoding="utf-8")
    NARRATIVE_PATH.write_text(narrative, encoding="utf-8")

    print("=" * 72)
    print("论文结果材料生成完成")
    print(f"- Markdown 表格: {MD_TABLE_PATH}")
    print(f"- LaTeX 表格:   {TEX_TABLE_PATH}")
    print(f"- 中文描述:     {NARRATIVE_PATH}")
    print("=" * 72)


if __name__ == "__main__":
    main()
```
