"""
最小实验市场服务画像补丁脚本

作用：
1. 将第二步初始化出来的 6 个服务映射到新的 6 个独立 service_key
2. 更新 capabilities.service_key
3. 同步更新 input_description
"""

from __future__ import annotations

import json
from typing import Dict

from sqlalchemy.orm import Session

from app.database import SessionLocal
from app.models.service import Service


SERVICE_KEY_MAPPING: Dict[str, str] = {
    "smart-summarizer-pro-01": "svc_txt_summary_pro_001",
    "budget-summarizer-lite-01": "svc_txt_summary_budget_001",
    "study-notes-summarizer-01": "svc_txt_summary_study_001",
    "report-summary-flex-01": "svc_txt_summary_flex_001",
    "json-extractor-pro-01": "svc_ext_json_pro_001",
    "json-extractor-budget-01": "svc_ext_json_budget_001"
}


def main() -> None:
    db: Session = SessionLocal()
    try:
        services = db.query(Service).filter(
            Service.name.in_(list(SERVICE_KEY_MAPPING.keys()))
        ).all()

        updated = 0
        for service in services:
            new_service_key = SERVICE_KEY_MAPPING.get(service.name)
            if not new_service_key:
                continue

            capabilities = dict(service.capabilities or {})
            capabilities["service_key"] = new_service_key
            service.capabilities = capabilities

            service.input_description = json.dumps({
                "service_key": new_service_key,
                "task_input": {
                    "text": "string"
                },
                "stage": "string"
            }, ensure_ascii=False)

            updated += 1

        db.commit()

        print("=" * 60)
        print(f"最小服务市场补丁完成，已更新 {updated} 个服务")
        for service in services:
            print(
                f"[Patched] {service.name} -> "
                f"{(service.capabilities or {}).get('service_key')}"
            )
        print("=" * 60)

    finally:
        db.close()


if __name__ == "__main__":
    main()

"""
最小闭环实验执行器

当前阶段目标：
1. 使用真实 HTTP 接口调用 AgentDNS discovery
2. 在 Top-K 候选中按 Agent 画像选择服务
3. 调用服务并获取 usage_id
4. 自动提交 review
5. 打印 trust summary 变化

当前仅支持：
- 质量优先型 Agent
- 成本优先型 Agent
- 简单摘要任务
- 简单 JSON 抽取任务
"""

from __future__ import annotations

import random
import time
from typing import Any, Dict, List, Optional

import requests


BASE_URL = "http://127.0.0.1:8000/api/v1"

AGENTS = {
    "quality": {
        "name": "quality-agent-01",
        "api_key": "agent_exp_quality_01",
        "profile": "quality_first"
    },
    "cost": {
        "name": "cost-agent-01",
        "api_key": "agent_exp_cost_01",
        "profile": "cost_first"
    }
}

TASKS: List[Dict[str, Any]] = [
    {

        "task_type": "summary",
        "query": "summary chinese document",
        "category": "text_generation",

        "top_k": 4
    },
    {
        "task_id": "task_sum_002",
        "task_type": "summary",
        "query": "study notes summary chinese",
        "category": "text_generation",
        "text": "这是一份机器学习课程复习笔记，包含监督学习、无监督学习、梯度下降和模型评估等内容。",
        "top_k": 4
    },
    {
        "task_id": "task_ext_001",
        "task_type": "json_extraction",
        "query": "json extraction structured fields",
        "category": "structured_extraction",
        "text": "姓名：张三；学号：2024001；课程：数据库系统；成绩：89。",
        "top_k": 2
    },
    {

        "category": "structured_extraction",
        "text": "订单号=A10293，客户=李四，金额=239.5元，状态=已支付。",
        "top_k": 2
    }
] * 3  # 共 12 个任务实例


def auth_headers(api_key: str) -> Dict[str, str]:
    """
    构造认证请求头
    """
    return {
        "Authorization": f"Bearer {api_key}",
        "Content-Type": "application/json"
    }


def search_services(api_key: str, task: Dict[str, Any]) -> List[Dict[str, Any]]:
    """
    调用 discovery 搜索服务
    """
    payload = {
        "query": task["query"],
        "category": task["category"],
        "limit": 10,
        "return_tool_format": False,
        "include_trust": True,
        "sort_by": "balanced"
    }

    resp = requests.post(
        f"{BASE_URL}/client/discovery/search",
        headers=auth_headers(api_key),
        json=payload,
        timeout=30
    )
    resp.raise_for_status()

    data = resp.json()
    return data.get("services", [])


def choose_service(profile: str, candidates: List[Dict[str, Any]], top_k: int) -> Dict[str, Any]:
    """
    在 Top-K 候选里按画像做最终选择

    说明：
    - 当前是最小实验策略，不引入复杂 utility
    - 只使用 discovery 返回的公开字段
    """
    visible = candidates[:top_k]
    if not visible:
        raise RuntimeError("Top-K 候选为空")

    def score_quality(service: Dict[str, Any]) -> float:
        trust = float(service.get("trust_score") or 0.0)
        price = float(service.get("price_per_unit") or 0.0)
        capabilities = service.get("capabilities") or {}
        role_type = capabilities.get("role_type", "")
        specialist_bonus = 8.0 if role_type == "specialist" else 0.0
        return trust + specialist_bonus - price * 20.0

    def score_cost(service: Dict[str, Any]) -> float:
        trust = float(service.get("trust_score") or 0.0)
        price = float(service.get("price_per_unit") or 0.0)
        return (10.0 - price * 100.0) + trust * 0.2

    if profile == "quality_first":
        ranked = sorted(visible, key=score_quality, reverse=True)
    elif profile == "cost_first":
        ranked = sorted(visible, key=score_cost, reverse=True)
    else:
        ranked = visible

    return ranked[0]


def call_service(api_key: str, service: Dict[str, Any], task: Dict[str, Any]) -> requests.Response:
    """
    调用客户端服务接口
    """
    capabilities = service.get("capabilities") or {}
    service_key = capabilities.get("service_key")
    if not service_key:
        raise RuntimeError(f"服务 {service.get('name')} 缺少 capabilities.service_key")

    payload = {
        "agentdns_url": service["agentdns_uri"],
        "input_data": {
            "service_key": service_key,
            "task_input": {
                "text": task["text"]
            },
            "stage": "cold_start"
        },
        "method": "POST"
    }

    resp = requests.post(
        f"{BASE_URL}/client/services/call",
        headers=auth_headers(api_key),
        json=payload,
        timeout=60
    )
    return resp


def extract_usage_id(resp: requests.Response) -> Optional[int]:
    """
    从响应头提取 usage_id
    """
    value = resp.headers.get("X-AgentDNS-Usage-ID")
    if not value:
        return None
    try:
        return int(value)
    except ValueError:
        return None


def build_review_payload(
    usage_id: int,
    service: Dict[str, Any],
    call_data: Dict[str, Any],
    profile: str
) -> Dict[str, Any]:
    """
    根据可见调用结果构造最小结构化评价

    说明：
    - 当前阶段不读取 oracle 日志
    - 只按可见 status 和价格做规则化评价
    """
    outcome = call_data.get("status", "fail")
    price = float(service.get("price_per_unit") or 0.0)

    if outcome == "success":
        task_fit = 5
        output_quality = 4 if profile == "quality_first" else 3
        protocol_adherence = 5
        would_reuse = True
    elif outcome == "partial":
        task_fit = 3
        output_quality = 3
        protocol_adherence = 4
        would_reuse = False
    else:
        task_fit = 1
        output_quality = 1
        protocol_adherence = 2
        would_reuse = False

    if price <= 0.03:
        cost_satisfaction = 5
    elif price <= 0.06:
        cost_satisfaction = 4
    elif price <= 0.08:
        cost_satisfaction = 3
    else:
        cost_satisfaction = 2

    return {
        "usage_id": usage_id,
        "outcome": outcome,
        "task_fit": task_fit,
        "output_quality": output_quality,
        "protocol_adherence": protocol_adherence,
        "would_reuse": would_reuse,
        "cost_satisfaction": cost_satisfaction,
        "feedback_text": f"minimal runner auto review for {service.get('name')}",
        "evidence": {
            "visible_status": outcome,
            "profile": profile
        }
    }


def submit_review(api_key: str, review_payload: Dict[str, Any]) -> Dict[str, Any]:
    """
    提交评价
    """
    resp = requests.post(
        f"{BASE_URL}/reviews/",
        headers=auth_headers(api_key),
        json=review_payload,
        timeout=30
    )
    resp.raise_for_status()
    return resp.json()


def get_trust_summary(api_key: str, service_id: int) -> Dict[str, Any]:
    """
    获取服务 trust 摘要
    """
    resp = requests.get(
        f"{BASE_URL}/reviews/services/{service_id}/summary",
        headers=auth_headers(api_key),
        timeout=30
    )
    resp.raise_for_status()
    return resp.json()


def run_single_task(agent_cfg: Dict[str, Any], task: Dict[str, Any]) -> None:
    """
    运行单个任务实例
    """
    api_key = agent_cfg["api_key"]
    profile = agent_cfg["profile"]

    print("\n" + "=" * 72)
    print(f"[TASK] {task['task_id']} | profile={profile} | type={task['task_type']}")

    candidates = search_services(api_key, task)
    if not candidates:
        print("[SEARCH] 未找到候选服务")
        return

    print(f"[SEARCH] 返回候选数: {len(candidates)}")

    selected = choose_service(profile, candidates, task["top_k"])
    print(
        f"[SELECT] name={selected['name']} "
        f"price={selected.get('price_per_unit')} "
        f"trust={selected.get('trust_score')} "
        f"service_key={(selected.get('capabilities') or {}).get('service_key')}"
    )

    resp = call_service(api_key, selected, task)
    try:
        call_data = resp.json()
    except Exception:
        print(f"[CALL] 响应无法解析为 JSON, status={resp.status_code}, body={resp.text[:200]}")
        return

    usage_id = extract_usage_id(resp)

    print(f"[CALL] http_status={resp.status_code} usage_id={usage_id}")
    print(f"[CALL] result_status={call_data.get('status')} request_id={call_data.get('request_id')}")

    if usage_id is None:
        print("[REVIEW] 未获取到 usage_id，无法提交评价")
        return

    review_payload = build_review_payload(
        usage_id=usage_id,
        service=selected,
        call_data=call_data,
        profile=profile
    )

    review_data = submit_review(api_key, review_payload)
    print(f"[REVIEW] review_id={review_data.get('id')} outcome={review_payload['outcome']}")

    trust_summary = get_trust_summary(api_key, selected["id"])
    print(
        f"[TRUST] service={selected['name']} "
        f"trust_score={trust_summary.get('trust_score')} "
        f"objective_score={trust_summary.get('objective_score')} "
        f"subjective_score={trust_summary.get('subjective_score')} "
        f"rating_count={trust_summary.get('rating_count')}"
    )


def main() -> None:
    """
    执行最小闭环实验
    """
    task_list = TASKS.copy()
    random.shuffle(task_list)

    agents = [
        AGENTS["quality"],
        AGENTS["cost"]
    ]

    for idx, task in enumerate(task_list, start=1):
        agent_cfg = agents[idx % len(agents)]
        try:
            run_single_task(agent_cfg, task)
        except Exception as e:
            print(f"[ERROR] task={task['task_id']} profile={agent_cfg['profile']} error={e}")
        time.sleep(0.5)

    print("\n" + "=" * 72)
    print("最小闭环实验完成")
    print("=" * 72)


if __name__ == "__main__":
    main()
