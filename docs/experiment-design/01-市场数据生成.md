## 1）`experiments/mock_services/registry.py`

```python
"""
模拟服务注册表

负责：
1. 维护 service_key -> 服务静态与行为配置 的映射
2. 为 mock service 层提供统一查询入口
"""

from __future__ import annotations

import json
from pathlib import Path
from typing import Any, Dict, Optional


class ServiceRegistry:
    """
    模拟服务注册表

    当前阶段先支持：
    - 从 JSON 文件加载服务配置
    - 按 service_key 查询服务配置

    后续可扩展：
    - YAML 加载
    - 多文件合并
    - 阶段覆盖规则
    """

    def __init__(self) -> None:
        # 内部服务配置映射：service_key -> config
        self._services: Dict[str, Dict[str, Any]] = {}

    def load_from_json(self, config_path: str | Path) -> None:
        """
        从 JSON 文件加载服务配置

        Args:
            config_path: JSON 配置文件路径
        """
        path = Path(config_path)
        if not path.exists():
            raise FileNotFoundError(f"服务配置文件不存在: {path}")

        with path.open("r", encoding="utf-8") as f:
            data = json.load(f)

        if not isinstance(data, list):
            raise ValueError("服务配置文件必须是列表格式")

        loaded_count = 0
        for item in data:
            if not isinstance(item, dict):
                continue

            service_key = item.get("service_key")
            if not service_key:
                continue

            self._services[service_key] = item
            loaded_count += 1

        if loaded_count == 0:
            raise ValueError("未加载到任何有效服务配置")

    def register_inline(self, config: Dict[str, Any]) -> None:
        """
        直接注册单个服务配置（用于最小闭环阶段）

        Args:
            config: 单个服务配置
        """
        service_key = config.get("service_key")
        if not service_key:
            raise ValueError("服务配置缺少 service_key")

        self._services[service_key] = config

    def get(self, service_key: str) -> Optional[Dict[str, Any]]:
        """
        获取服务配置

        Args:
            service_key: 服务唯一键

        Returns:
            服务配置；若不存在则返回 None
        """
        return self._services.get(service_key)

    def require(self, service_key: str) -> Dict[str, Any]:
        """
        获取服务配置；若不存在则抛错

        Args:
            service_key: 服务唯一键

        Returns:
            服务配置
        """
        config = self.get(service_key)
        if config is None:
            raise KeyError(f"未找到 service_key={service_key} 对应的服务配置")
        return config

    def count(self) -> int:
        """
        返回当前已注册服务数量
        """
        return len(self._services)

    def list_keys(self) -> list[str]:
        """
        返回全部 service_key 列表
        """
        return sorted(self._services.keys())
```

## 2）`experiments/mock_services/oracle_logger.py`

```python
"""
Oracle 日志记录器

负责：
1. 将每次模拟服务调用的隐藏真值写入 jsonl 文件
2. 后续供实验分析与评估使用
"""

from __future__ import annotations

import json
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict


class OracleLogger:
    """
    Oracle 日志记录器
    """

    def __init__(self, output_path: str | Path) -> None:
        self.output_path = Path(output_path)
        self.output_path.parent.mkdir(parents=True, exist_ok=True)

    def log(self, record: Dict[str, Any]) -> None:
        """
        追加写入一条 oracle 日志

        Args:
            record: 日志字典
        """
        safe_record = dict(record)
        safe_record.setdefault(
            "logged_at",
            datetime.now(timezone.utc).isoformat()
        )

        with self.output_path.open("a", encoding="utf-8") as f:
            f.write(json.dumps(safe_record, ensure_ascii=False) + "\n")
```

## 3）`experiments/mock_services/app.py`

```python
"""
模拟服务应用入口

当前阶段目标：
1. 提供最小可运行 FastAPI 服务
2. 支持按 service_key 查找服务配置
3. 返回最小模拟响应
4. 写 oracle 日志
"""

from __future__ import annotations

import random
import uuid
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict, Optional

from fastapi import FastAPI, HTTPException, Header
from pydantic import BaseModel, Field

from .registry import ServiceRegistry
from .oracle_logger import OracleLogger


# =========================
# 全局基础对象
# =========================

app = FastAPI(
    title="AgentDNS Experiments Mock Services",
    version="0.1.0",
    description="论文实验用高拟真模拟服务容器"
)

registry = ServiceRegistry()
oracle_logger = OracleLogger("experiments/outputs/oracle/oracle_calls.jsonl")


# =========================
# 请求/响应模型
# =========================

class MockServiceRequest(BaseModel):
    """
    模拟服务请求体

    说明：
    - service_key: 当前要调用的逻辑服务键
    - task_input: 任务输入，当前阶段暂不做复杂解析
    - request_id: 外部可传；若未传则内部生成
    - stage: 当前实验阶段，先保留字段
    """
    service_key: str = Field(..., description="逻辑服务唯一键")
    task_input: Dict[str, Any] = Field(default_factory=dict, description="任务输入")
    request_id: Optional[str] = Field(default=None, description="请求ID")
    stage: str = Field(default="cold_start", description="实验阶段")


class MockServiceResponse(BaseModel):
    """
    模拟服务响应体
    """
    request_id: str
    service_key: str
    status: str
    result: Dict[str, Any]
    mock_meta: Dict[str, Any]


# =========================
# 初始化最小内置服务
# =========================

def _bootstrap_minimal_services() -> None:
    """
    注册最小内置服务

    当前阶段只放两个：
    1. 文本摘要服务
    2. JSON 抽取服务
    """
    registry.register_inline({
        "service_key": "svc_txt_demo_summary_001",
        "category": "text_generation",
        "subcategory": "summarization",
        "role_type": "specialist",
        "quality_tier": "A",
        "stability_type": "stable",
        "http_mode": "sync",
        "behavior_profile": {
            "base_success_rate": 0.90,
            "base_partial_rate": 0.06,
            "base_fail_rate": 0.04,
            "schema_adherence_rate": 0.95,
            "constraint_follow_rate": 0.88,
            "content_quality_score_mean": 0.82,
            "content_quality_score_std": 0.08
        }
    })

    registry.register_inline({
        "service_key": "svc_ext_demo_json_001",
        "category": "structured_extraction",
        "subcategory": "json_extraction",
        "role_type": "specialist",
        "quality_tier": "A",
        "stability_type": "stable",
        "http_mode": "sync",
        "behavior_profile": {
            "base_success_rate": 0.93,
            "base_partial_rate": 0.04,
            "base_fail_rate": 0.03,
            "schema_adherence_rate": 0.97,
            "constraint_follow_rate": 0.92,
            "content_quality_score_mean": 0.86,
            "content_quality_score_std": 0.05
        }
    })


_bootstrap_minimal_services()


# =========================
# 辅助函数
# =========================

def _choose_outcome(config: Dict[str, Any]) -> str:
    """
    根据服务行为配置抽样本次调用结果

    Returns:
        success / partial / fail
    """
    profile = config.get("behavior_profile", {})
    success_rate = float(profile.get("base_success_rate", 0.8))
    partial_rate = float(profile.get("base_partial_rate", 0.1))
    fail_rate = float(profile.get("base_fail_rate", 0.1))

    total = success_rate + partial_rate + fail_rate
    if total <= 0:
        return "fail"

    # 归一化，避免配置误差
    success_rate /= total
    partial_rate /= total
    fail_rate /= total

    r = random.random()
    if r < success_rate:
        return "success"
    if r < success_rate + partial_rate:
        return "partial"
    return "fail"


def _sample_quality_score(config: Dict[str, Any]) -> float:
    """
    采样隐藏内容质量分数，范围裁剪到 [0, 1]
    """
    profile = config.get("behavior_profile", {})
    mean = float(profile.get("content_quality_score_mean", 0.75))
    std = float(profile.get("content_quality_score_std", 0.08))

    value = random.gauss(mean, std)
    return max(0.0, min(1.0, value))


def _sample_schema_valid(config: Dict[str, Any]) -> bool:
    """
    采样 schema 是否有效
    """
    profile = config.get("behavior_profile", {})
    rate = float(profile.get("schema_adherence_rate", 0.9))
    return random.random() < rate


def _build_mock_result(service_key: str, outcome: str, task_input: Dict[str, Any]) -> Dict[str, Any]:
    """
    构造最小业务结果

    当前阶段不做复杂任务区分，只按 service_key 前缀返回简单结果。
    """
    if service_key.startswith("svc_txt_"):
        source_text = str(task_input.get("text", ""))
        summary = source_text[:80] if source_text else "这是一个模拟摘要结果。"
        return {
            "text": f"摘要结果：{summary}",
            "status": outcome
        }

    if service_key.startswith("svc_ext_"):
        source_text = str(task_input.get("text", ""))
        return {
            "structured_data": {
                "source_length": len(source_text),
                "keywords": ["模拟", "抽取", "结果"]
            },
            "status": outcome
        }

    return {
        "message": "默认模拟服务响应",
        "status": outcome
    }


# =========================
# 路由
# =========================

@app.get("/health")
async def health_check() -> Dict[str, Any]:
    """
    健康检查接口
    """
    return {
        "status": "healthy",
        "service": "experiments-mock-services",
        "registered_services": registry.count()
    }


@app.get("/registry")
async def get_registry_summary() -> Dict[str, Any]:
    """
    查看当前已注册的最小服务列表
    """
    return {
        "count": registry.count(),
        "service_keys": registry.list_keys()
    }


@app.post("/mock/invoke", response_model=MockServiceResponse)
async def invoke_mock_service(
    request: MockServiceRequest,
    x_agentdns_request_id: Optional[str] = Header(default=None)
) -> MockServiceResponse:
    """
    通用模拟服务调用入口

    当前阶段只做最小闭环：
    - 按 service_key 找配置
    - 抽样 success / partial / fail
    - 生成最小业务输出
    - 写 oracle 日志
    """
    try:
        service_config = registry.require(request.service_key)
    except KeyError as e:
        raise HTTPException(status_code=404, detail=str(e)) from e

    request_id = request.request_id or x_agentdns_request_id or str(uuid.uuid4())

    outcome = _choose_outcome(service_config)
    quality_score = _sample_quality_score(service_config)
    schema_valid = _sample_schema_valid(service_config)

    result = _build_mock_result(
        service_key=request.service_key,
        outcome=outcome,
        task_input=request.task_input
    )

    # 写隐藏 oracle 日志
    oracle_logger.log({
        "request_id": request_id,
        "service_key": request.service_key,
        "category": service_config.get("category"),
        "subcategory": service_config.get("subcategory"),
        "role_type": service_config.get("role_type"),
        "quality_tier": service_config.get("quality_tier"),
        "stability_type": service_config.get("stability_type"),
        "stage": request.stage,
        "oracle_outcome": outcome,
        "oracle_content_quality_score": quality_score,
        "oracle_schema_valid": schema_valid,
        "oracle_constraint_satisfied": True,
        "is_nested_outer_call": False,
        "nested_subcall_count": 0,
        "received_at": datetime.now(timezone.utc).isoformat()
    })

    return MockServiceResponse(
        request_id=request_id,
        service_key=request.service_key,
        status=outcome,
        result=result,
        mock_meta={
            "category": service_config.get("category"),
            "subcategory": service_config.get("subcategory"),
            "http_mode": service_config.get("http_mode", "sync")
        }
    )
```
