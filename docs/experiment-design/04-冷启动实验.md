```python
"""
任务实例化器

负责：
1. 从最小任务模板文件加载种子模板
2. 按给定随机种子生成具体任务实例
3. 输出统一任务实例结构，供 runner 使用
"""

from __future__ import annotations

import json
import random
from pathlib import Path
from typing import Any, Dict, List


DEFAULT_TEMPLATE_PATH = Path("experiments/configs/tasks/minimal_task_templates.json")


def load_task_templates(template_path: str | Path = DEFAULT_TEMPLATE_PATH) -> List[Dict[str, Any]]:
    """
    加载任务模板列表
    """
    path = Path(template_path)
    if not path.exists():
        raise FileNotFoundError(f"任务模板文件不存在: {path}")

    with path.open("r", encoding="utf-8") as f:
        data = json.load(f)

    if not isinstance(data, list):
        raise ValueError("任务模板文件必须是列表格式")

    return data


def instantiate_tasks(
    total_count: int = 12,
    seed: int = 20260414,
    template_path: str | Path = DEFAULT_TEMPLATE_PATH
) -> List[Dict[str, Any]]:
    """
    根据模板实例化任务列表

    Args:
        total_count: 生成的任务实例总数
        seed: 固定随机种子
        template_path: 模板文件路径

    Returns:
        实例化后的任务列表
    """
    rng = random.Random(seed)
    templates = load_task_templates(template_path)

    if not templates:
        raise ValueError("任务模板为空，无法实例化任务")

    instances: List[Dict[str, Any]] = []

    for idx in range(total_count):
        template = templates[idx % len(templates)]
        input_candidates = template.get("input_candidates") or []
        if not input_candidates:
            raise ValueError(f"模板 {template.get('template_id')} 缺少 input_candidates")

        text = rng.choice(input_candidates)

        instance = {
            "task_instance_id": f"inst_{idx + 1:04d}",
            "template_id": template["template_id"],
            "domain": template["domain"],
            "complexity": template["complexity"],
            "task_type": template["task_type"],
            "query": template["query"],
            "category": template["category"],
            "top_k": template["top_k"],
            "text": text,
            "hard_constraints": template.get("hard_constraints", {}),
            "soft_preferences": template.get("soft_preferences", {}),
            "seed": seed
        }
        instances.append(instance)

    rng.shuffle(instances)
    return instances
```

## `experiments/runners/template_closed_loop_runner.py`

```python
"""
模板驱动最小闭环实验执行器

作用：
1. 从任务模板实例化任务
2. 调用 discovery -> call -> review -> trust
3. 将每个任务的关键结果落到 jsonl 运行日志
"""

from __future__ import annotations

import json
import time
from pathlib import Path
from typing import Any, Dict, List, Optional

import requests

from .task_instantiator import instantiate_tasks


BASE_URL = "http://127.0.0.1:8000/api/v1"
OUTPUT_PATH = Path("experiments/outputs/runs/minimal_closed_loop_runs.jsonl")

AGENTS = [
    {
        "agent_name": "quality-agent-01",
        "api_key": "agent_exp_quality_01",
        "profile": "quality_first"
    },
    {
        "agent_name": "cost-agent-01",
        "api_key": "agent_exp_cost_01",
        "profile": "cost_first"
    }
]


def auth_headers(api_key: str) -> Dict[str, str]:
    """
    构造认证头
    """
    return {
        "Authorization": f"Bearer {api_key}",
        "Content-Type": "application/json"
    }


def append_run_log(record: Dict[str, Any]) -> None:
    """
    追加写入一条运行日志
    """
    OUTPUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    with OUTPUT_PATH.open("a", encoding="utf-8") as f:
        f.write(json.dumps(record, ensure_ascii=False) + "\n")


def search_services(api_key: str, task: Dict[str, Any]) -> List[Dict[str, Any]]:
    """
    调用 discovery 接口
    """
    max_price = task.get("hard_constraints", {}).get("max_price")

    payload = {
        "query": task["query"],
        "category": task["category"],
        "limit": 10,
        "return_tool_format": False,
        "include_trust": True,
        "sort_by": "balanced"
    }

    if max_price is not None:
        payload["max_price"] = max_price

    resp = requests.post(
        f"{BASE_URL}/client/discovery/search",
        headers=auth_headers(api_key),
        json=payload,
        timeout=30
    )
    resp.raise_for_status()
    data = resp.json()
    return data.get("services", [])


def choose_service(profile: str, candidates: List[Dict[str, Any]], top_k: int) -> Dict[str, Any]:
    """
    在 Top-K 内做最终选择
    """
    visible = candidates[:top_k]
    if not visible:
        raise RuntimeError("Top-K 候选为空")

    def quality_score(svc: Dict[str, Any]) -> float:
        trust = float(svc.get("trust_score") or 0.0)
        price = float(svc.get("price_per_unit") or 0.0)
        capabilities = svc.get("capabilities") or {}
        role_type = capabilities.get("role_type", "")
        specialist_bonus = 8.0 if role_type == "specialist" else 0.0
        return trust + specialist_bonus - price * 20.0

    def cost_score(svc: Dict[str, Any]) -> float:
        trust = float(svc.get("trust_score") or 0.0)
        price = float(svc.get("price_per_unit") or 0.0)
        return (10.0 - price * 100.0) + trust * 0.2

    if profile == "quality_first":
        ranked = sorted(visible, key=quality_score, reverse=True)
    elif profile == "cost_first":
        ranked = sorted(visible, key=cost_score, reverse=True)
    else:
        ranked = visible

    return ranked[0]


def call_service(api_key: str, service: Dict[str, Any], task: Dict[str, Any]) -> requests.Response:
    """
    调用客户端服务接口
    """
    capabilities = service.get("capabilities") or {}
    service_key = capabilities.get("service_key")
    if not service_key:
        raise RuntimeError(f"服务 {service.get('name')} 缺少 capabilities.service_key")

    payload = {
        "agentdns_url": service["agentdns_uri"],
        "input_data": {
            "service_key": service_key,
            "task_input": {
                "text": task["text"]
            },
            "stage": "cold_start"
        },
        "method": "POST"
    }

    resp = requests.post(
        f"{BASE_URL}/client/services/call",
        headers=auth_headers(api_key),
        json=payload,
        timeout=60
    )
    return resp


def extract_usage_id(resp: requests.Response) -> Optional[int]:
    """
    从响应头中提取 usage_id
    """
    value = resp.headers.get("X-AgentDNS-Usage-ID")
    if not value:
        return None
    try:
        return int(value)
    except ValueError:
        return None


def build_review_payload(
    usage_id: int,
    service: Dict[str, Any],
    call_data: Dict[str, Any],
    profile: str
) -> Dict[str, Any]:
    """
    生成最小结构化评价
    """
    outcome = call_data.get("status", "fail")
    price = float(service.get("price_per_unit") or 0.0)

    if outcome == "success":
        task_fit = 5
        output_quality = 4 if profile == "quality_first" else 3
        protocol_adherence = 5
        would_reuse = True
    elif outcome == "partial":
        task_fit = 3
        output_quality = 3
        protocol_adherence = 4
        would_reuse = False
    else:
        task_fit = 1
        output_quality = 1
        protocol_adherence = 2
        would_reuse = False

    if price <= 0.03:
        cost_satisfaction = 5
    elif price <= 0.06:
        cost_satisfaction = 4
    elif price <= 0.08:
        cost_satisfaction = 3
    else:
        cost_satisfaction = 2

    return {
        "usage_id": usage_id,
        "outcome": outcome,
        "task_fit": task_fit,
        "output_quality": output_quality,
        "protocol_adherence": protocol_adherence,
        "would_reuse": would_reuse,
        "cost_satisfaction": cost_satisfaction,
        "feedback_text": f"template minimal runner review for {service.get('name')}",
        "evidence": {
            "visible_status": outcome,
            "profile": profile
        }
    }


def submit_review(api_key: str, payload: Dict[str, Any]) -> Dict[str, Any]:
    """
    提交评价
    """
    resp = requests.post(
        f"{BASE_URL}/reviews/",
        headers=auth_headers(api_key),
        json=payload,
        timeout=30
    )
    resp.raise_for_status()
    return resp.json()


def get_trust_summary(api_key: str, service_id: int) -> Dict[str, Any]:
    """
    获取服务 trust 摘要
    """
    resp = requests.get(
        f"{BASE_URL}/reviews/services/{service_id}/summary",
        headers=auth_headers(api_key),
        timeout=30
    )
    resp.raise_for_status()
    return resp.json()


def run_single_task(task: Dict[str, Any], agent_cfg: Dict[str, Any]) -> None:
    """
    执行单个任务，并写入运行日志
    """
    api_key = agent_cfg["api_key"]
    profile = agent_cfg["profile"]

    log_record: Dict[str, Any] = {
        "task_instance_id": task["task_instance_id"],
        "template_id": task["template_id"],
        "domain": task["domain"],
        "complexity": task["complexity"],
        "task_type": task["task_type"],
        "agent_name": agent_cfg["agent_name"],
        "profile": profile,
        "query": task["query"],
        "category": task["category"],
        "top_k": task["top_k"],
        "max_price": task.get("hard_constraints", {}).get("max_price"),
        "search_candidates_count": 0,
        "visible_candidates": [],
        "selected_service_name": None,
        "selected_service_id": None,
        "selected_service_price": None,
        "selected_service_trust_before": None,
        "call_http_status": None,
        "call_status": None,
        "usage_id": None,
        "review_id": None,
        "trust_score_after": None,
        "objective_score_after": None,
        "subjective_score_after": None,
        "rating_count_after": None,
        "error": None
    }

    try:
        candidates = search_services(api_key, task)
        log_record["search_candidates_count"] = len(candidates)
        log_record["visible_candidates"] = [
            {
                "name": svc.get("name"),
                "id": svc.get("id"),
                "price_per_unit": svc.get("price_per_unit"),
                "trust_score": svc.get("trust_score")
            }
            for svc in candidates[:task["top_k"]]
        ]

        if not candidates:
            log_record["error"] = "no_candidates"
            append_run_log(log_record)
            return

        selected = choose_service(profile, candidates, task["top_k"])
        log_record["selected_service_name"] = selected.get("name")
        log_record["selected_service_id"] = selected.get("id")
        log_record["selected_service_price"] = selected.get("price_per_unit")
        log_record["selected_service_trust_before"] = selected.get("trust_score")

        resp = call_service(api_key, selected, task)
        log_record["call_http_status"] = resp.status_code

        try:
            call_data = resp.json()
        except Exception:
            log_record["error"] = f"call_json_parse_error: {resp.text[:200]}"
            append_run_log(log_record)
            return

        log_record["call_status"] = call_data.get("status")
        usage_id = extract_usage_id(resp)
        log_record["usage_id"] = usage_id

        if usage_id is None:
            log_record["error"] = "missing_usage_id"
            append_run_log(log_record)
            return

        review_payload = build_review_payload(
            usage_id=usage_id,
            service=selected,
            call_data=call_data,
            profile=profile
        )
        review_data = submit_review(api_key, review_payload)
        log_record["review_id"] = review_data.get("id")

        trust_summary = get_trust_summary(api_key, selected["id"])
        log_record["trust_score_after"] = trust_summary.get("trust_score")
        log_record["objective_score_after"] = trust_summary.get("objective_score")
        log_record["subjective_score_after"] = trust_summary.get("subjective_score")
        log_record["rating_count_after"] = trust_summary.get("rating_count")

    except Exception as e:
        log_record["error"] = str(e)

    append_run_log(log_record)


def main() -> None:
    """
    执行模板驱动最小闭环实验
    """
    # 每次运行先清空旧日志，避免最小阶段数据混在一起
    if OUTPUT_PATH.exists():
        OUTPUT_PATH.unlink()

    tasks = instantiate_tasks(total_count=16, seed=20260414)

    for idx, task in enumerate(tasks):
        agent_cfg = AGENTS[idx % len(AGENTS)]
        print(
            f"[RUN] task={task['task_instance_id']} "
            f"type={task['task_type']} "
            f"profile={agent_cfg['profile']}"
        )
        run_single_task(task, agent_cfg)
        time.sleep(0.4)

    print("=" * 72)
    print(f"模板驱动最小闭环实验完成，日志输出: {OUTPUT_PATH}")
    print("=" * 72)


if __name__ == "__main__":
    main()
```

# 4B：最小结果汇总

```python
"""
最小实验结果汇总器

作用：
1. 读取 jsonl 运行日志
2. 输出最小统计摘要
3. 导出 csv 汇总表
"""

from __future__ import annotations

import csv
import json
from pathlib import Path
from typing import Any, Dict, List


INPUT_PATH = Path("experiments/outputs/runs/minimal_closed_loop_runs.jsonl")
CSV_OUTPUT_PATH = Path("experiments/outputs/runs/minimal_closed_loop_summary.csv")


def load_records(path: Path) -> List[Dict[str, Any]]:
    """
    加载 jsonl 记录
    """
    if not path.exists():
        raise FileNotFoundError(f"运行日志不存在: {path}")

    records: List[Dict[str, Any]] = []
    with path.open("r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            records.append(json.loads(line))
    return records


def export_csv(records: List[Dict[str, Any]], path: Path) -> None:
    """
    导出最小 csv 汇总
    """
    path.parent.mkdir(parents=True, exist_ok=True)

    fieldnames = [
        "task_instance_id",
        "template_id",
        "domain",
        "complexity",
        "task_type",
        "agent_name",
        "profile",
        "selected_service_name",
        "selected_service_price",
        "selected_service_trust_before",
        "call_http_status",
        "call_status",
        "usage_id",
        "review_id",
        "trust_score_after",
        "objective_score_after",
        "subjective_score_after",
        "rating_count_after",
        "error"
    ]

    with path.open("w", encoding="utf-8", newline="") as f:
        writer = csv.DictWriter(f, fieldnames=fieldnames)
        writer.writeheader()
        for record in records:
            writer.writerow({k: record.get(k) for k in fieldnames})


def main() -> None:
    """
    读取日志并打印最小统计摘要
    """
    records = load_records(INPUT_PATH)
    total = len(records)
    success_calls = sum(1 for r in records if r.get("call_status") == "success")
    partial_calls = sum(1 for r in records if r.get("call_status") == "partial")
    fail_calls = sum(1 for r in records if r.get("call_status") == "fail")
    reviewed = sum(1 for r in records if r.get("review_id") is not None)
    with_usage = sum(1 for r in records if r.get("usage_id") is not None)
    no_error = sum(1 for r in records if not r.get("error"))

    export_csv(records, CSV_OUTPUT_PATH)

    print("=" * 72)
    print("最小实验结果汇总")
    print(f"总任务数: {total}")
    print(f"成功调用数: {success_calls}")
    print(f"部分完成数: {partial_calls}")
    print(f"失败调用数: {fail_calls}")
    print(f"获取 usage_id 数: {with_usage}")
    print(f"成功提交 review 数: {reviewed}")
    print(f"无异常任务数: {no_error}")
    print(f"CSV 输出: {CSV_OUTPUT_PATH}")
    print("=" * 72)


if __name__ == "__main__":
    main()
```
