## 新建文件

```
experiments/scripts/run_v1_v3_coldstart.py
"""
V1 vs V3 冷启动最小对比实验一键执行脚本

执行流程：
1. 清理旧冷启动输出
2. 重置最小实验状态
3. 运行 V1（无 warm-up）
4. 再次重置
5. 运行 V3（无 warm-up）
6. 生成冷启动汇总 CSV
7. 将冷启动结果复制为独立快照文件

说明：
- 当前使用已有的 v1_v3_paired_runner.py
- 不执行 trust warm-up
"""

from __future__ import annotations

import shutil
import subprocess
import sys
from pathlib import Path


PROJECT_ROOT = Path(__file__).resolve().parents[2]
RUNS_DIR = PROJECT_ROOT / "experiments/outputs/runs"


def run_step(title: str, cmd: list[str]) -> None:
    print("\n" + "=" * 80)
    print(f"[STEP] {title}")
    print(f"[CMD] {' '.join(cmd)}")
    print("=" * 80)

    result = subprocess.run(
        cmd,
        cwd=PROJECT_ROOT,
        text=True
    )
    if result.returncode != 0:
        raise RuntimeError(f"步骤失败: {title}")


def safe_delete(path: Path) -> None:
    if path.exists():
        path.unlink()
        print(f"[DELETE] {path}")
    else:
        print(f"[SKIP] {path} 不存在")


def main() -> None:
    python_exec = sys.executable

    print("\n" + "=" * 80)
    print("[STEP] 清理旧冷启动输出")
    print("=" * 80)

    safe_delete(RUNS_DIR / "v1_vs_v3_runs.jsonl")
    safe_delete(RUNS_DIR / "v1_vs_v3_summary.csv")
    safe_delete(RUNS_DIR / "v1_vs_v3_coldstart_runs.jsonl")
    safe_delete(RUNS_DIR / "v1_vs_v3_coldstart_summary.csv")

    run_step(
        "重置最小实验状态（V1 冷启动前）",
        [python_exec, "experiments/seeds/reset_minimal_experiment_state.py"]
    )

    run_step(
        "运行 V1 冷启动 evaluation",
        [python_exec, "experiments/runners/v1_v3_paired_runner.py", "V1"]
    )

    run_step(
        "重置最小实验状态（V3 冷启动前）",
        [python_exec, "experiments/seeds/reset_minimal_experiment_state.py"]
    )

    run_step(
        "运行 V3 冷启动 evaluation",
        [python_exec, "experiments/runners/v1_v3_paired_runner.py", "V3"]
    )

    run_step(
        "生成冷启动汇总 CSV",
        [python_exec, "experiments/analyzers/v1_v3_summary.py"]
    )

    # 保存为独立快照，避免后续被预热版本覆盖
    src_runs = RUNS_DIR / "v1_vs_v3_runs.jsonl"
    src_summary = RUNS_DIR / "v1_vs_v3_summary.csv"

    dst_runs = RUNS_DIR / "v1_vs_v3_coldstart_runs.jsonl"
    dst_summary = RUNS_DIR / "v1_vs_v3_coldstart_summary.csv"

    shutil.copyfile(src_runs, dst_runs)
    shutil.copyfile(src_summary, dst_summary)

    print("\n" + "=" * 80)
    print("V1 vs V3 冷启动对比实验已完成")
    print(f"- {dst_runs}")
    print(f"- {dst_summary}")
    print("=" * 80)


if __name__ == "__main__":
    main()
```

现在再新建一个脚本，专门把：

- `v1_vs_v3_summary.csv`（当前预热后版本）

## 新建文件

```
experiments/analyzers/build_dual_stage_report.py
"""
生成“冷启动 vs 预热后”双阶段论文结果材料

输入：
- experiments/outputs/runs/v1_vs_v3_coldstart_summary.csv
- experiments/outputs/runs/v1_vs_v3_summary.csv   （当前默认视为预热后版本）

输出：
- experiments/outputs/reports/v1_v3_dual_stage_table.md
- experiments/outputs/reports/v1_v3_dual_stage_table.tex
- experiments/outputs/reports/v1_v3_dual_stage_narrative.md
"""

from __future__ import annotations

import csv
from pathlib import Path
from typing import Dict, List


REPORT_DIR = Path("experiments/outputs/reports")

COLDSTART_PATH = Path("experiments/outputs/runs/v1_vs_v3_coldstart_summary.csv")
WARMED_PATH = Path("experiments/outputs/runs/v1_vs_v3_summary.csv")

MD_PATH = REPORT_DIR / "v1_v3_dual_stage_table.md"
TEX_PATH = REPORT_DIR / "v1_v3_dual_stage_table.tex"
NARRATIVE_PATH = REPORT_DIR / "v1_v3_dual_stage_narrative.md"


def load_csv(path: Path) -> List[Dict[str, str]]:
    if not path.exists():
        raise FileNotFoundError(f"缺少输入文件: {path}")
    with path.open("r", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        return list(reader)


def to_map(rows: List[Dict[str, str]]) -> Dict[str, Dict[str, str]]:
    result: Dict[str, Dict[str, str]] = {}
    for row in rows:
        version = row.get("version")
        if version:
            result[version] = row
    return result


def f_ratio_percent(x: str | float) -> str:
    return f"{float(x) * 100:.1f}%"


def f4(x: str | float) -> str:
    return f"{float(x):.4f}"


def build_md(
    cold_v1: Dict[str, str],
    cold_v3: Dict[str, str],
    warm_v1: Dict[str, str],
    warm_v3: Dict[str, str]
) -> str:
    return f"""# 冷启动 vs 预热后：V1 与 V3 双阶段结果表

## 1. 冷启动阶段

| 指标 | V1 | V3 |
|------|----|----|
| 任务成功率 | {f_ratio_percent(cold_v1["task_success_rate"])} | {f_ratio_percent(cold_v3["task_success_rate"])} |
| 平均 Oracle Utility | {f4(cold_v1["avg_oracle_utility"])} | {f4(cold_v3["avg_oracle_utility"])} |
| 平均总成本 | {f4(cold_v1["avg_total_price"])} | {f4(cold_v3["avg_total_price"])} |

## 2. 预热后阶段

| 指标 | V1 | V3 |
|------|----|----|
| 任务成功率 | {f_ratio_percent(warm_v1["task_success_rate"])} | {f_ratio_percent(warm_v3["task_success_rate"])} |
| 平均 Oracle Utility | {f4(warm_v1["avg_oracle_utility"])} | {f4(warm_v3["avg_oracle_utility"])} |
| 平均总成本 | {f4(warm_v1["avg_total_price"])} | {f4(warm_v3["avg_total_price"])} |

## 3. 结论概括

- 冷启动阶段，V1 与 V3 差异有限，说明在缺乏历史交互信息时，trust 机制尚未形成明显优势。
- 预热后阶段，V3 开始在任务成功率、Oracle Utility 或成本控制上体现出区别于 V1 的行为特征。
- 这表明 trust 机制的收益依赖于服务市场中逐渐积累的使用与评价信号。
"""


def build_tex(
    cold_v1: Dict[str, str],
    cold_v3: Dict[str, str],
    warm_v1: Dict[str, str],
    warm_v3: Dict[str, str]
) -> str:
    return rf"""\begin{{table}}[htbp]
\centering
\caption{{冷启动阶段与预热后阶段中 V1 与 V3 的对比结果}}
\label{{tab:v1_v3_dual_stage}}
\begin{{tabular}}{{llcc}}
\hline
阶段 & 指标 & V1 & V3 \\
\hline
冷启动 & 任务成功率 & {float(cold_v1["task_success_rate"]) * 100:.1f}\% & {float(cold_v3["task_success_rate"]) * 100:.1f}\% \\
冷启动 & 平均 Oracle Utility & {float(cold_v1["avg_oracle_utility"]):.4f} & {float(cold_v3["avg_oracle_utility"]):.4f} \\
冷启动 & 平均总成本 & {float(cold_v1["avg_total_price"]):.4f} & {float(cold_v3["avg_total_price"]):.4f} \\
\hline
预热后 & 任务成功率 & {float(warm_v1["task_success_rate"]) * 100:.1f}\% & {float(warm_v3["task_success_rate"]) * 100:.1f}\% \\
预热后 & 平均 Oracle Utility & {float(warm_v1["avg_oracle_utility"]):.4f} & {float(warm_v3["avg_oracle_utility"]):.4f} \\
预热后 & 平均总成本 & {float(warm_v1["avg_total_price"]):.4f} & {float(warm_v3["avg_total_price"]):.4f} \\
\hline
\end{{tabular}}
\end{{table}}
"""


def build_narrative(
    cold_v1: Dict[str, str],
    cold_v3: Dict[str, str],
    warm_v1: Dict[str, str],
    warm_v3: Dict[str, str]
) -> str:
    cold_v1_success = float(cold_v1["task_success_rate"])
    cold_v3_success = float(cold_v3["task_success_rate"])
    warm_v1_success = float(warm_v1["task_success_rate"])
    warm_v3_success = float(warm_v3["task_success_rate"])

    cold_v1_util = float(cold_v1["avg_oracle_utility"])
    cold_v3_util = float(cold_v3["avg_oracle_utility"])
    warm_v1_util = float(warm_v1["avg_oracle_utility"])
    warm_v3_util = float(warm_v3["avg_oracle_utility"])

    cold_v1_cost = float(cold_v1["avg_total_price"])
    cold_v3_cost = float(cold_v3["avg_total_price"])
    warm_v1_cost = float(warm_v1["avg_total_price"])
    warm_v3_cost = float(warm_v3["avg_total_price"])

    return f"""# 冷启动 vs 预热后 双阶段结果描述草稿

为了分析 trust 机制在不同平台状态下的作用，本文进一步将最小对比实验划分为冷启动阶段与预热后阶段进行观察。冷启动阶段中，平台尚未形成明显的历史评价与信誉差异，此时 V1 与 V3 的结果接近：V1 的任务成功率为 {cold_v1_success * 100:.1f}%，V3 为 {cold_v3_success * 100:.1f}%；V1 的平均 Oracle Utility 为 {cold_v1_util:.4f}，V3 为 {cold_v3_util:.4f}；两者在平均总成本上也较为接近（V1 为 {cold_v1_cost:.4f}，V3 为 {cold_v3_cost:.4f}）。这一现象说明，在缺乏历史交互信号时，trust 机制尚未形成足以显著影响服务选择的差异化信息。

在经过 warm-up 预热后，服务之间开始积累不同的 usage 与 review 信息，trust 分层逐步形成。此时，V1 与 V3 的差异开始显现：V1 的任务成功率为 {warm_v1_success * 100:.1f}%，V3 为 {warm_v3_success * 100:.1f}%；V1 的平均 Oracle Utility 为 {warm_v1_util:.4f}，V3 为 {warm_v3_util:.4f}；平均总成本则分别为 {warm_v1_cost:.4f} 与 {warm_v3_cost:.4f}。与冷启动阶段相比，预热后阶段更能体现 trust 对服务选择的影响，即系统不再仅依据词面召回或简单排序，而会逐渐利用已有评价信号，在质量、成本与适配性之间形成更具平台意义的平衡。

因此，这组双阶段实验表明：trust 机制的有效性并不是静态成立的，而是依赖于服务市场中的持续交互与评价积累。当服务市场处于冷启动状态时，trust 的作用有限；而当平台中逐步形成稳定的历史信号后，trust 才开始对服务选择和整体实验结果产生更实质的影响。
"""


def main() -> None:
    REPORT_DIR.mkdir(parents=True, exist_ok=True)

    cold_map = to_map(load_csv(COLDSTART_PATH))
    warm_map = to_map(load_csv(WARMED_PATH))

    for name, mapping in [("coldstart", cold_map), ("warmed", warm_map)]:
        if "V1" not in mapping or "V3" not in mapping:
            raise ValueError(f"{name} 结果中必须同时包含 V1 与 V3")

    cold_v1, cold_v3 = cold_map["V1"], cold_map["V3"]
    warm_v1, warm_v3 = warm_map["V1"], warm_map["V3"]

    MD_PATH.write_text(build_md(cold_v1, cold_v3, warm_v1, warm_v3), encoding="utf-8")
    TEX_PATH.write_text(build_tex(cold_v1, cold_v3, warm_v1, warm_v3), encoding="utf-8")
    NARRATIVE_PATH.write_text(build_narrative(cold_v1, cold_v3, warm_v1, warm_v3), encoding="utf-8")

    print("=" * 72)
    print("双阶段论文结果材料生成完成")
    print(f"- Markdown 表格: {MD_PATH}")
    print(f"- LaTeX 表格:   {TEX_PATH}")
    print(f"- 中文描述:     {NARRATIVE_PATH}")
    print("=" * 72)


if __name__ == "__main__":
    main()
```
