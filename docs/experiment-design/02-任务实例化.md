很好，第二步开始。

这一步只做一个目标：

> 让 AgentDNS 平台里真正出现一批**可发现、可调用**的最小实验服务。

这一轮我建议先**不用 HTTP 注册组织/登录/建服务**，而是先用**数据库 seed 脚本**直接落盘。原因很简单：

- 这一步是实验基座初始化，不是论文主实验
- 你现在还没把 discovery runner 和调用 runner 接起来
- 直接 seed 能更快、更稳地得到一个最小服务市场
- 后面真正跑实验时，调用和评价仍然走真实 HTTP 链路，不影响论文主线

------

# 第二步的目标状态

做完这一步后，你的平台里应该最少有：

- 2 个服务提供者用户
- 2 个组织
- 2 个客户端用户
- 4 个 Agent
- 6 个公开服务

并且这 6 个服务都指向你刚刚跑通的 mock service 容器：

- `http://127.0.0.1:9001/mock/invoke`

注意，这 6 个服务虽然 endpoint 相同，但它们会通过：

- `service_key`
- `input_description`
- `capabilities`
- `tags`

表现成**不同逻辑服务实例**。

    db: Session,
    username: str,
    email: str,
    role: str,
    password: str,
    full_name: str,
    balance: float = 200.0
) -> User:
    """
    确保用户存在；若不存在则创建

    Args:
        db: 数据库会话
        username: 用户名

        role: 角色

        balance: 初始余额

    if user:
        return user

        username=username,
        email=email,
        full_name=full_name,
        hashed_password=get_password_hash(password),
        role=role,
        is_active=True,
        is_verified=True,
        balance=balance

    db.add(user)
    db.commit()
    db.refresh(user)
    return user


def ensure_organization(
    db: Session,

    name: str,

    description: str

    """

        display_name=display_name,
        description=description,
        owner_id=owner.id,
        is_verified=True
    )
    db.add(org)
    db.commit()
    db.refresh(org)
    return org

        user_id=user.id,
        is_active=True,
        is_suspended=False

def ensure_service(
    db: Session,
    org: Organization,
    payload: Dict[str, Any]
) -> Service:
    """
    确保服务存在；若不存在则创建
    """
    agentdns_uri = f"agentdns://{payload['agentdns_path']}"

    service = db.query(Service).filter(
        Service.agentdns_uri == agentdns_uri
    ).first()
    if service:
        return service

    service = Service(
        name=payload["name"],
        category=payload["category"],
        agentdns_uri=agentdns_uri,
        description=payload["description"],
        version=payload["version"],
        is_active=True,
        is_public=True,
        endpoint_url=MOCK_ENDPOINT,
        protocol="HTTP",
        authentication_required=True,
        pricing_model="per_request",
        price_per_unit=payload["price_per_unit"],
        currency="USD",
        tags=payload["tags"],
        capabilities=payload["capabilities"],
        organization_id=org.id,
        agentdns_path=payload["agentdns_path"],
        http_method=payload["http_method"],
        http_mode=payload["http_mode"],
        input_description=payload["input_description"],
        output_description=payload["output_description"],
        service_api_key=None
    )

    db.add(service)
    db.commit()
    db.refresh(service)

    metadata = ServiceMetadata(
        service_id=service.id,
        search_keywords=payload["tags"],
        status="active",
        examples={
            "input": {
                "service_key": payload["capabilities"]["service_key"],
                "task_input": {
                    "text": "这是一个用于实验初始化的测试输入。"
                },
                "stage": "cold_start"
            },
            "output": {
                "status": "success"
            }
        }
    )
    db.add(metadata)

    trust_stats = ServiceTrustStats(
        service_id=service.id,
        trust_score=0.0,
        objective_score=0.0,
        subjective_score=0.0,
        success_rate=0.0,
        avg_response_time_ms=0.0,
        rating_count=0,
        usage_count=0
    )
    db.add(trust_stats)

    db.commit()
    db.refresh(service)
    return service
