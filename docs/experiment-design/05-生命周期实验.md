```python
"""
任务成功判定器

当前阶段目标：
1. 对中等任务的两步执行结果进行整体成功判定
2. 输出 task_success / failure_reason
"""

from __future__ import annotations

from typing import Any, Dict


def evaluate_medium_task_success(run_record: Dict[str, Any]) -> Dict[str, Any]:
    """
    评估中等任务是否成功

    判定逻辑（最小版）：
    1. 两步都必须成功拿到 usage_id
    2. 两步调用状态都必须为 success 或 partial
    3. 最终步骤不能为 fail
    4. 若总成本超过硬约束，则直接失败

    Args:
        run_record: 中等任务单条运行日志

    Returns:
        包含 task_success 与 failure_reason 的字典
    """
    step1 = run_record.get("step1", {})
    step2 = run_record.get("step2", {})
    hard_constraints = run_record.get("hard_constraints", {})

    max_total_price = hard_constraints.get("max_total_price")
    total_cost = float(run_record.get("total_price_estimate") or 0.0)

    if step1.get("usage_id") is None:
        return {
            "task_success": False,
            "failure_reason": "step1_missing_usage"
        }

    if step2.get("usage_id") is None:
        return {
            "task_success": False,
            "failure_reason": "step2_missing_usage"
        }

    if step1.get("call_status") not in {"success", "partial"}:
        return {
            "task_success": False,
            "failure_reason": "step1_failed"
        }

    if step2.get("call_status") not in {"success", "partial"}:
        return {
            "task_success": False,
            "failure_reason": "step2_failed"
        }

    if max_total_price is not None and total_cost > float(max_total_price):
        return {
            "task_success": False,
            "failure_reason": "cost_exceeded"
        }

    return {
        "task_success": True,
        "failure_reason": None
    }
```

```python
"""
Oracle Utility 评估器（最小版）

当前阶段目标：
1. 从 oracle_calls.jsonl 中按 request_id 找到隐藏真值
2. 对两步任务计算最小组合 utility
"""

from __future__ import annotations

import json
from pathlib import Path
from typing import Any, Dict, Optional


ORACLE_LOG_PATH = Path("experiments/outputs/oracle/oracle_calls.jsonl")


def _load_oracle_index() -> Dict[str, Dict[str, Any]]:
    """
    加载 oracle 日志并按 request_id 建索引
    """
    if not ORACLE_LOG_PATH.exists():
        raise FileNotFoundError(f"oracle 日志不存在: {ORACLE_LOG_PATH}")

    index: Dict[str, Dict[str, Any]] = {}
    with ORACLE_LOG_PATH.open("r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            record = json.loads(line)
            request_id = record.get("request_id")
            if request_id:
                index[request_id] = record
    return index


def _normalize_cost(cost_value: float, max_expected: float = 0.20) -> float:
    """
    将成本转换为 [0,1] 的收益分，越便宜越高
    """
    if max_expected <= 0:
        return 1.0
    ratio = min(max(cost_value / max_expected, 0.0), 1.0)
    return 1.0 - ratio


def _normalize_latency(latency_ms: float, max_expected_ms: float = 5000.0) -> float:
    """
    将时延转换为 [0,1] 的收益分，越快越高
    """
    if max_expected_ms <= 0:
        return 1.0
    ratio = min(max(latency_ms / max_expected_ms, 0.0), 1.0)
    return 1.0 - ratio


def evaluate_medium_oracle_utility(run_record: Dict[str, Any]) -> Dict[str, Any]:
    """
    计算中等任务的最小 oracle utility

    当前最小版采用四维：
    - 质量
    - 成本
    - 时延
    - 约束满足

    权重暂按当前模板软偏好近似映射：
    quality_priority / cost_priority / latency_priority
    约束满足额外固定 0.15 权重，并对前三项做归一化缩放
    """
    oracle_index = _load_oracle_index()

    step1 = run_record.get("step1", {})
    step2 = run_record.get("step2", {})
    soft_preferences = run_record.get("soft_preferences", {})

    request_id_1 = step1.get("request_id")
    request_id_2 = step2.get("request_id")

    oracle_1: Optional[Dict[str, Any]] = oracle_index.get(request_id_1) if request_id_1 else None
    oracle_2: Optional[Dict[str, Any]] = oracle_index.get(request_id_2) if request_id_2 else None

    if not oracle_1 or not oracle_2:
        return {
            "oracle_utility": None,
            "oracle_quality_score": None,
            "oracle_cost_score": None,
            "oracle_latency_score": None,
            "oracle_constraint_score": None,
            "oracle_missing": True
        }

    quality_priority = float(soft_preferences.get("quality_priority", 0.5))
    cost_priority = float(soft_preferences.get("cost_priority", 0.25))
    latency_priority = float(soft_preferences.get("latency_priority", 0.25))

    # 约束满足给固定额外权重
    constraint_weight = 0.15
    scale = 1.0 - constraint_weight

    raw_sum = quality_priority + cost_priority + latency_priority
    if raw_sum <= 0:
        quality_w = 0.5 * scale
        cost_w = 0.25 * scale
        latency_w = 0.25 * scale
    else:
        quality_w = quality_priority / raw_sum * scale
        cost_w = cost_priority / raw_sum * scale
        latency_w = latency_priority / raw_sum * scale

    quality_score = (
        float(oracle_1.get("oracle_content_quality_score", 0.0)) +
        float(oracle_2.get("oracle_content_quality_score", 0.0))
    ) / 2.0

    cost_total = float(step1.get("selected_service_price") or 0.0) + float(step2.get("selected_service_price") or 0.0)
    cost_score = _normalize_cost(cost_total)

    latency_total = float(step1.get("oracle_latency_ms") or 0.0) + float(step2.get("oracle_latency_ms") or 0.0)
    latency_score = _normalize_latency(latency_total)

    constraint_flags = [
        bool(oracle_1.get("oracle_constraint_satisfied", False)),
        bool(oracle_2.get("oracle_constraint_satisfied", False))
    ]
    constraint_score = 1.0 if all(constraint_flags) else 0.0

    utility = (
        quality_score * quality_w +
        cost_score * cost_w +
        latency_score * latency_w +
        constraint_score * constraint_weight
    )

    return {
        "oracle_utility": round(utility, 6),
        "oracle_quality_score": round(quality_score, 6),
        "oracle_cost_score": round(cost_score, 6),
        "oracle_latency_score": round(latency_score, 6),
        "oracle_constraint_score": round(constraint_score, 6),
        "oracle_missing": False
    }
```

```python
"""

   - json_then_summary

from __future__ import annotations

import json
import random
import time
from pathlib import Path
from typing import Any, Dict, List, Optional

import requests

from experiments.evaluators.task_success import evaluate_medium_task_success
from experiments.evaluators.oracle_utility import evaluate_medium_oracle_utility


BASE_URL = "http://127.0.0.1:8000/api/v1"
TEMPLATE_PATH = Path("experiments/configs/tasks/medium_task_templates.json")
OUTPUT_PATH = Path("experiments/outputs/runs/medium_closed_loop_runs.jsonl")

AGENTS = [

def auth_headers(api_key: str) -> Dict[str, str]:
    return {
        "Authorization": f"Bearer {api_key}",
        "Content-Type": "application/json"
    }


def append_run_log(record: Dict[str, Any]) -> None:
    OUTPUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    with OUTPUT_PATH.open("a", encoding="utf-8") as f:
        f.write(json.dumps(record, ensure_ascii=False) + "\n")


def load_templates() -> List[Dict[str, Any]]:
    with TEMPLATE_PATH.open("r", encoding="utf-8") as f:
        return json.load(f)


def instantiate_medium_tasks(total_count: int = 8, seed: int = 20260414) -> List[Dict[str, Any]]:
    rng = random.Random(seed)
    templates = load_templates()

    instances: List[Dict[str, Any]] = []
    for idx in range(total_count):
        tpl = templates[idx % len(templates)]
        text = rng.choice(tpl["input_candidates"])
        instances.append({
            "task_instance_id": f"mid_{idx + 1:04d}",
            "template_id": tpl["template_id"],
            "domain": tpl["domain"],
            "complexity": tpl["complexity"],
            "workflow_type": tpl["workflow_type"],
            "query_step_1": tpl["query_step_1"],
            "category_step_1": tpl["category_step_1"],
            "query_step_2": tpl["query_step_2"],
            "category_step_2": tpl["category_step_2"],
            "top_k_step_1": tpl["top_k_step_1"],
            "top_k_step_2": tpl["top_k_step_2"],
            "hard_constraints": tpl.get("hard_constraints", {}),
            "soft_preferences": tpl.get("soft_preferences", {}),
            "text": text,
            "seed": seed
        })

    rng.shuffle(instances)
    return instances


def search_services(api_key: str, query: str, category: str, max_price: Optional[float]) -> List[Dict[str, Any]]:
    payload = {
        "query": query,
        "category": category,
        "limit": 10,
        "return_tool_format": False,
        "include_trust": True,
        "sort_by": "balanced"
    }
    if max_price is not None:
        payload["max_price"] = max_price

    resp = requests.post(
        f"{BASE_URL}/client/discovery/search",
        headers=auth_headers(api_key),
        json=payload,
        timeout=30
    )
    resp.raise_for_status()
    data = resp.json()
    return data.get("services", [])


def choose_service(profile: str, candidates: List[Dict[str, Any]], top_k: int) -> Dict[str, Any]:
    visible = candidates[:top_k]
    if not visible:
        raise RuntimeError("Top-K 候选为空")

    def quality_score(svc: Dict[str, Any]) -> float:
        trust = float(svc.get("trust_score") or 0.0)
        price = float(svc.get("price_per_unit") or 0.0)
        role_type = (svc.get("capabilities") or {}).get("role_type", "")
        specialist_bonus = 8.0 if role_type == "specialist" else 0.0
        return trust + specialist_bonus - price * 20.0

    def cost_score(svc: Dict[str, Any]) -> float:
        trust = float(svc.get("trust_score") or 0.0)
        price = float(svc.get("price_per_unit") or 0.0)
        return (10.0 - price * 100.0) + trust * 0.2

    ranked = visible
    if profile == "quality_first":
        ranked = sorted(visible, key=quality_score, reverse=True)
    elif profile == "cost_first":
        ranked = sorted(visible, key=cost_score, reverse=True)

    return ranked[0]


def call_service(api_key: str, service: Dict[str, Any], text: str) -> requests.Response:
    capabilities = service.get("capabilities") or {}
    service_key = capabilities.get("service_key")
    if not service_key:
        raise RuntimeError(f"服务 {service.get('name')} 缺少 capabilities.service_key")

    payload = {
        "agentdns_url": service["agentdns_uri"],
        "input_data": {
            "service_key": service_key,
            "task_input": {
                "text": text
            },
            "stage": "cold_start"
        },
        "method": "POST"
    }

    return requests.post(
        f"{BASE_URL}/client/services/call",
        headers=auth_headers(api_key),
        json=payload,
        timeout=60
    )


def extract_usage_id(resp: requests.Response) -> Optional[int]:
    value = resp.headers.get("X-AgentDNS-Usage-ID")
    if not value:
        return None
    try:
        return int(value)
    except ValueError:
        return None


def build_review_payload(usage_id: int, service: Dict[str, Any], call_data: Dict[str, Any], profile: str) -> Dict[str, Any]:
    outcome = call_data.get("status", "fail")
    price = float(service.get("price_per_unit") or 0.0)

    if outcome == "success":
        task_fit = 5
        output_quality = 4 if profile == "quality_first" else 3
        protocol_adherence = 5
        would_reuse = True
    elif outcome == "partial":
        task_fit = 3
        output_quality = 3
        protocol_adherence = 4
        would_reuse = False
    else:
        task_fit = 1
        output_quality = 1
        protocol_adherence = 2
        would_reuse = False

    if price <= 0.03:
        cost_satisfaction = 5
    elif price <= 0.06:
        cost_satisfaction = 4
    elif price <= 0.08:
        cost_satisfaction = 3
    else:
        cost_satisfaction = 2

    return {
        "usage_id": usage_id,
        "outcome": outcome,
        "task_fit": task_fit,
        "output_quality": output_quality,
        "protocol_adherence": protocol_adherence,
        "would_reuse": would_reuse,
        "cost_satisfaction": cost_satisfaction,
        "feedback_text": f"medium runner auto review for {service.get('name')}",
        "evidence": {
            "visible_status": outcome,
            "profile": profile
        }
    }


def submit_review(api_key: str, payload: Dict[str, Any]) -> Dict[str, Any]:

        f"{BASE_URL}/reviews/",
        headers=auth_headers(api_key),

def get_trust_summary(api_key: str, service_id: int) -> Dict[str, Any]:
    resp = requests.get(

        headers=auth_headers(api_key),

    resp.raise_for_status()

def execute_step(
    api_key: str,
    profile: str,
    query: str,
    category: str,
    top_k: int,
    input_text: str,
    max_price: Optional[float]
) -> Dict[str, Any]:
    step_log: Dict[str, Any] = {
        "query": query,
        "category": category,
        "top_k": top_k,
        "search_candidates_count": 0,
        "visible_candidates": [],
        "selected_service_name": None,
        "selected_service_id": None,
        "selected_service_price": None,
        "selected_service_trust_before": None,
        "call_http_status": None,
        "call_status": None,
        "request_id": None,
        "usage_id": None,
        "review_id": None,
        "trust_score_after": None,
        "objective_score_after": None,
        "subjective_score_after": None,
        "rating_count_after": None,
        "oracle_latency_ms": None,
        "error": None
    }

    try:
        candidates = search_services(api_key, query, category, max_price)
        step_log["search_candidates_count"] = len(candidates)
        step_log["visible_candidates"] = [
            {
                "name": svc.get("name"),
                "id": svc.get("id"),
                "price_per_unit": svc.get("price_per_unit"),
                "trust_score": svc.get("trust_score")
            }
            for svc in candidates[:top_k]
        ]

        if not candidates:
            step_log["error"] = "no_candidates"
            return step_log

        selected = choose_service(profile, candidates, top_k)
        step_log["selected_service_name"] = selected.get("name")
        step_log["selected_service_id"] = selected.get("id")
        step_log["selected_service_price"] = selected.get("price_per_unit")
        step_log["selected_service_trust_before"] = selected.get("trust_score")

        resp = call_service(api_key, selected, input_text)
        step_log["call_http_status"] = resp.status_code

        try:
            call_data = resp.json()
        except Exception:
            step_log["error"] = f"call_json_parse_error: {resp.text[:200]}"
            return step_log

        step_log["call_status"] = call_data.get("status")
        step_log["request_id"] = call_data.get("request_id")

        usage_id = extract_usage_id(resp)
        step_log["usage_id"] = usage_id
        if usage_id is None:
            step_log["error"] = "missing_usage_id"
            return step_log

        review_payload = build_review_payload(
            usage_id=usage_id,
            service=selected,
            call_data=call_data,
            profile=profile
        )
        review_data = submit_review(api_key, review_payload)
        step_log["review_id"] = review_data.get("id")

        trust_summary = get_trust_summary(api_key, selected["id"])
        step_log["trust_score_after"] = trust_summary.get("trust_score")
        step_log["objective_score_after"] = trust_summary.get("objective_score")
        step_log["subjective_score_after"] = trust_summary.get("subjective_score")
        step_log["rating_count_after"] = trust_summary.get("rating_count")

        # 当前阶段先从 response/result 推导下一步输入
        if "result" in call_data and isinstance(call_data["result"], dict):
            result_obj = call_data["result"]
            if "text" in result_obj:
                step_log["next_input_text"] = result_obj["text"]
            elif "structured_data" in result_obj:
                step_log["next_input_text"] = json.dumps(result_obj["structured_data"], ensure_ascii=False)
            else:
                step_log["next_input_text"] = json.dumps(result_obj, ensure_ascii=False)
        else:
            step_log["next_input_text"] = input_text

    except Exception as e:
        step_log["error"] = str(e)

    return step_log


def run_single_medium_task(task: Dict[str, Any], agent_cfg: Dict[str, Any]) -> None:
    api_key = agent_cfg["api_key"]
    profile = agent_cfg["profile"]
    max_total_price = task.get("hard_constraints", {}).get("max_total_price")

    run_record: Dict[str, Any] = {
        "task_instance_id": task["task_instance_id"],
        "template_id": task["template_id"],
        "domain": task["domain"],
        "complexity": task["complexity"],
        "workflow_type": task["workflow_type"],
        "agent_name": agent_cfg["agent_name"],
        "profile": profile,
        "hard_constraints": task.get("hard_constraints", {}),
        "soft_preferences": task.get("soft_preferences", {}),
        "step1": {},
        "step2": {},
        "total_price_estimate": 0.0,
        "task_success": None,
        "failure_reason": None,
        "oracle_utility": None,
        "oracle_quality_score": None,
        "oracle_cost_score": None,
        "oracle_latency_score": None,
        "oracle_constraint_score": None,
        "oracle_missing": None,
        "error": None
    }

    try:
        step1 = execute_step(
            api_key=api_key,
            profile=profile,
            query=task["query_step_1"],
            category=task["category_step_1"],
            top_k=task["top_k_step_1"],
            input_text=task["text"],
            max_price=max_total_price
        )
        run_record["step1"] = step1

        next_input_text = step1.get("next_input_text") or task["text"]

        step2 = execute_step(
            api_key=api_key,
            profile=profile,
            query=task["query_step_2"],
            category=task["category_step_2"],
            top_k=task["top_k_step_2"],
            input_text=next_input_text,
            max_price=max_total_price
        )
        run_record["step2"] = step2

        total_price = float(step1.get("selected_service_price") or 0.0) + float(step2.get("selected_service_price") or 0.0)
        run_record["total_price_estimate"] = round(total_price, 6)

        success_eval = evaluate_medium_task_success(run_record)
        run_record.update(success_eval)

        utility_eval = evaluate_medium_oracle_utility(run_record)
        run_record.update(utility_eval)

    except Exception as e:
        run_record["error"] = str(e)

    append_run_log(run_record)
```

```python
"""
中等任务实验结果汇总器
"""

from __future__ import annotations

import csv
import json
from pathlib import Path
from typing import Any, Dict, List


INPUT_PATH = Path("experiments/outputs/runs/medium_closed_loop_runs.jsonl")
CSV_OUTPUT_PATH = Path("experiments/outputs/runs/medium_closed_loop_summary.csv")


def load_records(path: Path) -> List[Dict[str, Any]]:
    if not path.exists():
        raise FileNotFoundError(f"运行日志不存在: {path}")

    records: List[Dict[str, Any]] = []
    with path.open("r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            records.append(json.loads(line))
    return records


def export_csv(records: List[Dict[str, Any]], path: Path) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)

    fieldnames = [
        "task_instance_id",
        "template_id",
        "domain",
        "complexity",
        "workflow_type",
        "agent_name",
        "profile",
        "task_success",
        "failure_reason",
        "total_price_estimate",
        "oracle_utility",
        "oracle_quality_score",
        "oracle_cost_score",
        "oracle_latency_score",
        "oracle_constraint_score",
        "oracle_missing",
        "step1_service",
        "step1_status",
        "step1_usage_id",
        "step2_service",
        "step2_status",
        "step2_usage_id",
        "error"
    ]

    with path.open("w", encoding="utf-8", newline="") as f:
        writer = csv.DictWriter(f, fieldnames=fieldnames)
        writer.writeheader()

        for record in records:
            writer.writerow({
                "task_instance_id": record.get("task_instance_id"),
                "template_id": record.get("template_id"),
                "domain": record.get("domain"),
                "complexity": record.get("complexity"),
                "workflow_type": record.get("workflow_type"),
                "agent_name": record.get("agent_name"),
                "profile": record.get("profile"),
                "task_success": record.get("task_success"),
                "failure_reason": record.get("failure_reason"),
                "total_price_estimate": record.get("total_price_estimate"),
                "oracle_utility": record.get("oracle_utility"),
                "oracle_quality_score": record.get("oracle_quality_score"),
                "oracle_cost_score": record.get("oracle_cost_score"),
                "oracle_latency_score": record.get("oracle_latency_score"),
                "oracle_constraint_score": record.get("oracle_constraint_score"),
                "oracle_missing": record.get("oracle_missing"),
                "step1_service": (record.get("step1") or {}).get("selected_service_name"),
                "step1_status": (record.get("step1") or {}).get("call_status"),
                "step1_usage_id": (record.get("step1") or {}).get("usage_id"),
                "step2_service": (record.get("step2") or {}).get("selected_service_name"),
                "step2_status": (record.get("step2") or {}).get("call_status"),
                "step2_usage_id": (record.get("step2") or {}).get("usage_id"),
                "error": record.get("error")
            })


def main() -> None:
    records = load_records(INPUT_PATH)

    total = len(records)
    success_tasks = sum(1 for r in records if r.get("task_success") is True)
    failed_tasks = sum(1 for r in records if r.get("task_success") is False)
    utility_available = sum(1 for r in records if r.get("oracle_utility") is not None)

    avg_utility = 0.0
    utility_values = [float(r["oracle_utility"]) for r in records if r.get("oracle_utility") is not None]
    if utility_values:
        avg_utility = sum(utility_values) / len(utility_values)

    export_csv(records, CSV_OUTPUT_PATH)

    print("=" * 72)
    print("中等任务实验结果汇总")
    print(f"总任务数: {total}")
    print(f"任务成功数: {success_tasks}")
    print(f"任务失败数: {failed_tasks}")
    print(f"可计算 oracle utility 数: {utility_available}")
    print(f"平均 oracle utility: {avg_utility:.4f}")
    print(f"CSV 输出: {CSV_OUTPUT_PATH}")
    print("=" * 72)


if __name__ == "__main__":
    main()
```
