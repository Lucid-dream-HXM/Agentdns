```python
"""
V1 vs V3 预热后最小对比实验一键执行脚本

执行流程：
1. 重置最小实验状态
2. 执行 warm-up
3. 跑 V1 evaluation
4. 再次重置
5. 再次 warm-up
6. 跑 V3 evaluation
7. 汇总 csv
8. 生成图表

说明：
- 该脚本默认在项目根目录执行
- 适合当前最小实验市场（10 个服务）复现实验
"""

from __future__ import annotations

import subprocess
import sys
from pathlib import Path


PROJECT_ROOT = Path(__file__).resolve().parents[2]


def run_step(title: str, cmd: list[str]) -> None:
    """
    执行单个步骤；失败则立即中断
    """
    print("\n" + "=" * 80)
    print(f"[STEP] {title}")
    print(f"[CMD] {' '.join(cmd)}")
    print("=" * 80)

    result = subprocess.run(
        cmd,
        cwd=PROJECT_ROOT,
        text=True
    )
    if result.returncode != 0:
        raise RuntimeError(f"步骤失败: {title}")


def remove_old_outputs() -> None:
    """
    删除旧的对比输出，避免混入旧结果
    """
    target_files = [
        PROJECT_ROOT / "experiments/outputs/runs/v1_vs_v3_runs.jsonl",
        PROJECT_ROOT / "experiments/outputs/runs/v1_vs_v3_summary.csv",
        PROJECT_ROOT / "experiments/outputs/figures/v1_v3_task_success_rate.png",
        PROJECT_ROOT / "experiments/outputs/figures/v1_v3_avg_oracle_utility.png",
        PROJECT_ROOT / "experiments/outputs/figures/v1_v3_avg_total_cost.png",
    ]

    print("\n" + "=" * 80)
    print("[STEP] 清理旧输出")
    print("=" * 80)

    for path in target_files:
        if path.exists():
            path.unlink()
            print(f"[DELETE] {path}")
        else:
            print(f"[SKIP] {path} 不存在")


def main() -> None:
    python_exec = sys.executable

    remove_old_outputs()

    run_step(
        "重置最小实验状态（V1 前）",
        [python_exec, "experiments/seeds/reset_minimal_experiment_state.py"]
    )

    run_step(
        "执行 warm-up（V1 前）",
        [python_exec, "experiments/runners/trust_warmup_runner.py"]
    )

    run_step(
        "运行 V1 evaluation",
        [python_exec, "experiments/runners/v1_v3_paired_runner.py", "V1"]
    )

    run_step(
        "重置最小实验状态（V3 前）",
        [python_exec, "experiments/seeds/reset_minimal_experiment_state.py"]
    )

    run_step(
        "执行 warm-up（V3 前）",
        [python_exec, "experiments/runners/trust_warmup_runner.py"]
    )

    run_step(
        "运行 V3 evaluation",
        [python_exec, "experiments/runners/v1_v3_paired_runner.py", "V3"]
    )

    run_step(
        "生成 V1 vs V3 汇总 CSV",
        [python_exec, "experiments/analyzers/v1_v3_summary.py"]
    )

    run_step(
        "生成 V1 vs V3 图表",
        [python_exec, "experiments/analyzers/plot_v1_v3_comparison.py"]
    )

    print("\n" + "=" * 80)
    print("V1 vs V3 预热后最小对比实验已全部完成")
    print("输出文件：")
    print("- experiments/outputs/runs/v1_vs_v3_runs.jsonl")
    print("- experiments/outputs/runs/v1_vs_v3_summary.csv")
    print("- experiments/outputs/figures/v1_v3_task_success_rate.png")
    print("- experiments/outputs/figures/v1_v3_avg_oracle_utility.png")
    print("- experiments/outputs/figures/v1_v3_avg_total_cost.png")
    print("=" * 80)


if __name__ == "__main__":
    main()
```
